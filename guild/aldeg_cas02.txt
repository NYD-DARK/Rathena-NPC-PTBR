-	script	Hohenschwangau#1::OutsideFlagsA2	722,{

	set .@GID, GetCastleData("aldeg_cas02",1);

	if (.@GID == 0) {
		mes "[ Edito do Divino Reino de Rune-Midgard ]";
		mes " ";
		mes "1. Siga a ordenança do Divino Reino de Rune-Midgard,";
		mes "Declaramos que";
		mes "năo há um mestre formal deste castelo.";
		mes " ";
		mes "2. Ŕquele que puder";
		mes "superar todas as provaçőes";
		mes "e destruir o Emperium,";
		mes "o rei o concederá";
		mes "a posse deste castelo.";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ Voz Ecoante ]";
		mes "Bravos guerreiros...";
		mes "Desejam retornar ao seu lugar honroso?";
		next;
		if(select("Return to the guild castle.:Quit.") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("aldeg_cas02",1)) {
				warp "aldeg_cas02",220,190;
				end;
			}
			end;
		}
		close;
	}
	mes "[ Edito do Divino Reino de Rune-Midgard ]";
	mes " ";
	mes "1. Siga a ordenança do Divino Reino de Rune-Midgard,";
	mes "aprovamos que este lugar está em";
	mes "the private prossession of ^ff0000"+GetGuildName(.@GID)+"^000000 Guild.";
	mes " ";
	mes "2. The guild Master of ^ff0000"+GetGuildName(.@GID)+"^000000 Guild is";
	mes "^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "Se houver alguém que se oponha a isto,";
	mes "prove sua força e honra com uma lâmina de aço em sua măo.";
	close;

OnRecvCastleA02:
	FlagEmblem GetCastleData("aldeg_cas02",1);
	end;
}
