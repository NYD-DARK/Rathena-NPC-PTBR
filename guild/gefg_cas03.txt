// Guild Flag (Outside Castle)
//============================================================
-	script	Yesnelph#1::OutsideFlagsG3	722,{

	set .@GID, GetCastleData("gefg_cas03",1);

	if (.@GID == 0) {
		mes "[ Édito do Reino de Rune-Midgard ]";
		mes " ";
		mes "1. Siga as ordenanças do Reino de Rune-Midgard, ";
		mes "Declaramos que";
		mes "năo há um mestre formal deste castelo.";
		mes " ";
		mes "2. Ŕquele que puder ";
		mes "superar todas as provaçőes";
		mes "e destruir o Emperium,";
		mes "o rei concederá a ele";
		mes "a posse deste castelo.";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ Voz Ecoante ]";
		mes "Braves...";
		mes "Desejam retornar ao seu lugar honroso?";
		next;
		if(select("Retornar ao castelo da guilda.:Sair.") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("gefg_cas03",1)) {
				warp "gefg_cas03",221,30;
				end;
			}
			end;
		}
		close;
	}
	mes "[ Édito do Reino de Rune-Midgard ]";
	mes " ";
	mes "1. Siga as ordenanças do Reino de Rune-Midgard, ";
	mes "aprovamos que este lugar está sob";
	mes "a posse privada da ^ff0000"+GetGuildName(.@GID)+"^000000 Guild.";
	mes " ";
	mes "2. O Mestre de Guilda da ^ff0000"+GetGuildName(.@GID)+"^000000 Guild is";
	mes "^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "Se houver alguém que se oponha a isto,";
	mes "prove sua força e honra com uma lâmina de aço em suas măos.";
	close;

OnRecvCastleG03:
	FlagEmblem GetCastleData("gefg_cas03",1);
	end;
}
