	callfunc "F_AuctionWarper",4;

function	script	F_AuctionWarper	{
	mes "[Guia da Casa de Leilőes]";
	mes "Olá, vocę gostaria";
	mes "de entrar na";
	mes "Casa de Leilőes?";
	next;
	if (select("Sim:Năo") == 1) {
		.@num = getarg(0);
		if (.@num == 1 || .@num == 4) {
			mes "[Guia da Casa de Leilőes]";
			mes "Ótimo! Entăo,";
			mes "espero que vocę se divirta";
			mes "e aproveite o leilăo~";
		} else {
			mes "[Guia da Casa de Leilőes]";
			mes "Aproveite o leilăo.";
		}
		close2;
		switch(.@num) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[Guia da Casa de Leilőes]";
	mes "Muito bem entăo,";
	mes "nos vemos mais tarde.";
	mes "Se vocę mudar de";
	mes "ideia, por favor, venha";
	mes "e aproveite os leilőes~";
	close;
}
