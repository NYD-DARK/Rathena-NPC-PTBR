xmas,140,137,3	script	Lutie Guide#xmas	717,{
	mes "[Guia de Lutie]";
	mes "Bem-vindo a ^8B4513Lutie^000000,";
	mes "a Vila da Neve.";
	mes "Você precisa de ajuda para navegar pela vila?";
	F_Navi("[Lutie Guide]");
	while (1) {
		next;
		switch(select("[ Loja de Presentes ]:[ Igreja ]:[ Loja de Armas e Armaduras ]:[ Loja de Utilidades ]:[ Casa do Papai Noel ]:[ Fábrica de Brinquedos ]:Remover marcas do minimapa:Cancelar")) {
		case 1:
			callsub L_Mark, F_Navi("Gift Shop","xmas,172,131","^008080");
			viewpoint 1,172,131,2,0x008080;
			break;
		case 2:
			callsub L_Mark, F_Navi("Church","xmas,104,287","^006400");
			viewpoint 1,104,287,1,0xAAFF00;
			break;
		case 3:
			callsub L_Mark, F_Navi("Weapon & Armor Shop","xmas,171,158","^FF1493");
			viewpoint 1,171,158,3,0xFF1493;
			break;
		case 4:
			callsub L_Mark, F_Navi("Tool Shop","xmas,122,131","^8B4513");
			viewpoint 1,122,131,4,0x8B4513;
			break;
		case 5:
			callsub L_Mark, F_Navi("Santa's House","xmas,149,237","^B9062F"),1;
			viewpoint 1,149,237,5,0x9400D3;
			break;
		case 6:
			callsub L_Mark, F_Navi("Toy Factory","xmas,143,312","^9400D3");
			viewpoint 1,143,312,6,0xFF0000;
			break;
		case 7:
			mes "[Guia de Lutie]";
			mes "Claro, eu removerei todas as marcas do seu minimapa.";
			mes "Há mais alguma coisa que eu possa fazer por você?";
			viewpoint 2,1,1,0,0xFFFFFF;
			viewpoint 2,1,1,1,0xFFFFFF;
			viewpoint 2,1,1,2,0xFFFFFF;
			viewpoint 2,1,1,3,0xFFFFFF;
			viewpoint 2,1,1,4,0xFFFFFF;
			viewpoint 2,1,1,5,0xFFFFFF;
			viewpoint 2,1,1,6,0xFFFFFF;
			break;
		case 8:
			mes "[Guia de Lutie]";
			mes "Tenha uma boa viagem.";
			close;
		}
	}
	end;
L_Mark:
	mes "[Guia de Lutie]";
	mes "Deixe-me marcar a localização do "+((getarg(1,0))?"":"the");
	mes getarg(0)+"^000000";
	mes "no seu minimapa.";
	mes "Você gostaria de verificar outros locais?";
	return;
}
