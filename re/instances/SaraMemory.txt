dali,139,118,4	script	Dimensional Device#sara	10007,{
	if (BaseLevel < 99) {
		mes "[Aventureiro Leon]";
		mes "Ei! Vocę é muito jovem para brincar com isso! Volte quando tiver pelo menos Nível 99!";
		close;
	}
	if ((saramemory < 1) || (saramemory == 2) || (saramemory == 3) || (saramemory == 5)) {
		mes "[Aventureiro Leon]";
		mes "Antes de vocę sair brincando com isso; Por que vocę năo vem aqui e fala comigo?";
		close;
	}

	// 20 hour cooldown
	set .@ins_sara_time,checkquest(15002,PLAYTIME);
	if (.@ins_sara_time == -1) {

		set .@party_id,getcharid(1);
		set .@p_name$,getpartyname(.@party_id);
		set .@md_name$,"Sara's Memories";

		if (!.@party_id) {
			mes "[Aventureiro Leon]";
			mes "Antes de entrar, vocę precisa organizar um grupo! Vocę năo sabe como? Digite /organizar NOMEDOPARTYQUIAQUI.";
			close;
		}

		if (is_party_leader() == true)
			set .@menu$, "Boot up the dimensional device.:Use the dimensional device.:Cancel.";
		else
			set .@menu$, ":Use the dimensional device.:Cancel.";

		switch(select(.@menu$)) {
		case 1:
			if (instance_create(.@md_name$) < 0) {
				mes "Party Name: "+.@p_name$;
				mes "Party Leader: "+strcharinfo(0);
				mes "^0000ff"+.@md_name$+" ^000000- Reservation Failed!";
				close;
			}
			mes "O dispositivo dimensional inicia corretamente. Use o dispositivo para entrar na fenda no espaço e no tempo.";
			close;
		case 2:
			switch(instance_enter(.@md_name$)) {
			case IE_OTHER:
				mes "Ocorreu um erro desconhecido.";
				close;
			case IE_NOINSTANCE:
				mes "The memorial dungeon "+.@md_name$+" does not exist.";
				mes "O líder do grupo ainda năo gerou a masmorra.";
				close;
			case IE_NOMEMBER:
				mes "Only the registered members can enter the instance "+.@md_name$+".";
				close;
			case IE_OK:
				mapannounce "dali",strcharinfo(0)+", member of the party "+.@p_name$+" entered the instance "+.@md_name$+".",bc_map,"0x00ff99";
				setquest 15002;
				end;
			}
		case 3:
			mes "O líder do grupo năo iniciou o dispositivo dimensional.";
			close;
		}
	} else if (.@ins_sara_time == 0 || .@ins_sara_time == 1) {
		mes "[Aventureiro Leon]";
		mes "Eu acho que em cerca de 20 horas, após sua última entrada na fenda, ela estará estável o suficiente para vocę atravessá-la novamente.";
		close;
	}
	else {
		erasequest 15002;
		mes "[Memórias de Sara] Cooldown foi limpo. Agora é possível usar o dispositivo dimensional novamente.";
		close;
	}
}
