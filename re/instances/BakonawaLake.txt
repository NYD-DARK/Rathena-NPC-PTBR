ma_scene01,174,179,4	script	Taho	541,{
	disable_items;
	if (checkweight(1201,1) == 0) {
		mes "Vocę está carregando itens demais. Volte depois de remover alguns deles.";
		close;
	}
	if (MaxWeight - Weight < 10000) {
		mes "Vocę está carregando muito peso. Volte depois de reduzir o peso.";
		close;
	}
	if (BaseLevel < 140) {
		mes "[Taho]";
		mes "Ei. Este é um lugar perigoso demais para vocę. Vocę năo conseguirá se manter seguro. Saia daqui e vá para a vila agora mesmo!";
		close;
	}
	if (malaya_bakona1 < 7 && malaya_bakona2 < 15) {
		mes "[Taho]";
		mes "Todos nós seremos mortos por aquele Bakonawa. Ah, nós devemos derrotá-lo, mas, infelizmente, năo estamos preparados.";
		close;
	}

	set .@party_id,getcharid(1);
	set .@md_name$, "Bakonawa Lake";

	set .@baku_time, checkquest(12278,PLAYTIME);
	if (.@baku_time == -1) {
		if (checkquest(12279) == -1) setquest 12279;
	}
	if (!.@party_id) {
		mes "[Taho]";
		mes "Năo existe nenhuma companhia que possa ir com vocę? Ou mesmo que queira ir sozinho, venha depois de organizar um grupo.";
		close;
	}
	if (.@baku_time == -1) {
		if (is_party_leader() == true) {
			mes "[Taho]";
			mes "Vocę é o líder dos caçadores de Bakonawa? A estrada năo é boa, entăo precisamos descer o penhasco com uma corda, tudo bem?";
			next;
			set .@i, select("Please weave a rope.:Now I will go down.:Cancel.");
		} else {
			mes "[Taho]";
			mes "Vocę veio para caçar o Bakonawa. O líder do seu grupo precisa me ajudar a trançar a corda para descermos.";
			next;
			set .@i, select(":Now I will go down.:Cancel.");
		}
		switch(.@i) {
		case 1:
			if (instance_create(.@md_name$) < 0) {
				mes "[Taho]";
				mes "Oh, a corda caiu. Tenho que fazer uma nova.";
				close;
			}
			mes "[Taho]";
			mes "Agora estou trançando, entăo vocę pode descer quando eu terminar.";
			close;
		case 2:
			switch(instance_enter(.@md_name$)) {
			case IE_OTHER:
				mes "[Taho]";
				mes "Ocorreu um erro desconhecido.";
				close;
			case IE_NOINSTANCE:
				mes "[Taho]";
				mes "A corda ainda năo foi trançada. Espere um pouco.";
				close;
			case IE_NOMEMBER:
				mes "[Taho]";
				mes "Civis năo podem entrar.";
				close;
			case IE_OK:
				mapannounce "ma_scene01","A party member, "+strcharinfo(0)+" of the party "+getpartyname(.@party_id)+" is entering the dungeon, "+.@md_name$+".",bc_map,"0x00ff99"; //FW_NORMAL 12 0 0
				setquest 12278;
				//warp "1@ma_b",64,51;
				close;
			}
		case 3:
			close;
		}
	} else if (.@baku_time == 0 || .@baku_time == 1) {
		mes "[Taho]";
		mes "A corda usada para chegar ao lago já está quebrada. Estamos limpando os arredores agora, entăo volte depois de algum tempo.";
		close;
	} else if (.@baku_time == 2) {
		mes "[Taho]";
		mes "Está tudo limpo por aqui. Agora vocę pode descer novamente após pegar uma corda.";
		erasequest 12278;
		close;
	}
}
