alberta_in,53,43,6	script	Merchant#mer	86,{
	if (Upper == 1) {
		if (Class == Job_Novice_High && (ADVJOB == Job_Whitesmith || ADVJOB == Job_Creator)) {
			mes "[Chefe Mahnsoo]";
			mes "Quanto tempo!";
			mes "Ei, vocę năo desistiu";
			mes "do seu negócio, năo é?";
			mes "O que aconteceu?";
			next;
			mes "[Chefe Mahnsoo]";
			mes "Uau...";
			mes "Vocę realmente esteve em Valhalla?! Uau, vocę percorreu um longo caminho...";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Chefe Mahnsoo]";
				mes "Hmmm...";
				mes "Parece que vocę năo está pronto para se tornar um Mercador novamente. Vá terminar de aprender as Habilidades de Aprendiz básicas primeiro.";
				next;
				mes "[Chefe Mahnsoo]";
				mes "Năo se preocupe, sempre teremos uma vaga de Mercador aberta para vocę. Apenas volte quando estiver pronto, está bem?";
				close;
			}
			mes "[Chefe Mahnsoo]";
			mes "Acho que é destino nos encontrarmos assim mais uma vez. Muito bem. Mais uma vez, deixe-me transformá-lo em um Mercador!";
			next;
			skill "NV_TRICKDEAD",0,SKILL_PERM;
			jobchange Job_Merchant_High;
			skill "MC_CARTREVOLUTION",1,SKILL_PERM;
			skill "MC_CHANGECART",1,SKILL_PERM;
			skill "MC_LOUD",1,SKILL_PERM;
			mes "[Chefe Mahnsoo]";
			mes "Ah~ Que nostálgico. Como nos velhos tempos! Tudo bem, dę o seu melhor!";
			close;
		}else{
			mes "[Chefe Mahnsoo]";
			mes "^333333*Suspiro*^000000";
			mes "Estou tăo entediado...";
			mes "Quando terei notícias da minha adorável Blossom?";
			close;
		}
	}
	mes "[Membro da Guilda Mahnsoo]";
	mes "Ei, por que vocę está aqui?";
	next;
	switch(select("I want to be a merchant.:I want to know more about merchants.:Ask him the requirements to be a merchant.:Nothing.")) {
	case 1:
		mes "[Membro da Guilda Mahnsoo]";
		if(BaseJob != Job_Novice) {
			if(BaseJob == Job_Merchant) {
				mes ".....? Desculpe? O que vocę está dizendo?";
				mes "Vocę já é um mercador. Ah, meu Deus..";
				mes "Hă?! ...Eu preciso rir agora?!";
			}else{
				mes "Ahh? Vocę está tentando dos dois jeitos?";
				mes "Que tal apenas se dedicar ŕ sua classe original?";
				next;
				mes "[Membro da Guilda Mahnsoo]";
				mes "Nós temos ética profissional, sabe.";
			}
			close;
		}else{
			mes "Quer ser um mercador? Hmm...";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Membro da Guilda Mahnsoo]";
				mes "Mas se vocę quer ser um mercador, seu nível de habilidade básica deve chegar ao Nível 9 ou vocę deve gastar todos os seus pontos de habilidade.";
				next;
				mes "[Membro da Guilda Mahnsoo]";
				mes "Vocę năo acha que precisamos aprender algumas habilidades básicas, embora apenas lidemos com dinheiro?";
				close;
			}
			mes "[Membro da Guilda Mahnsoo]";
			mes "Bom, acho que vocę está totalmente pronto para isso, vendo que seu nível de habilidade básica foi atingido.";
			mes "Agora eu permito que vocę seja um mercador.";
			next;
			mes "[Membro da Guilda Mahnsoo]";
			mes "Parabéns por se tornar um mercador!";
			mes "Parabéns novamente por ser um membro da guilda dos mercadores e um de nossos colegas. Espero sua participaçăo ativa a partir de agora!";
			callfunc "Job_Change",Job_Merchant;
			getitem 1381,1;	// N_Battle_Axe
			next;
			mes "[Membro da Guilda Mahnsoo]";
			mes "Com certeza, precisamos de jovens que tenham paixăo para alcançar nosso grande objetivo de garantir 20% do volume de moeda mundial. Entendeu? Hă?";
			mes "Bem, estou apenas dizendo... significa vamos ganhar muito dinheiro no final das contas. Vocęs sabem disso~";
			close;
		}
	case 2:
		mes "[Membro da Guilda Mahnsoo]";
		mes "Mercador? Para simplificar, a pessoa que vende produtos e ganha dinheiro é um mercador.";
		mes "Năo é bom em lutar e năo tem habilidades especiais de ataque/recuperaçăo... mas um mercador pode comprar coisas a um preço baixo e depois vendę-las e ganhar dinheiro.";
		next;
		mes "[Membro da Guilda Mahnsoo]";
		mes "Bem, um mercador tem uma habilidade suprema chamada Mammonita que atinge um inimigo com seu dinheiro... Podemos equipar tudo, exceto Arcos, Cajados e Espadas de Duas Măos. Mas sempre podemos vender e comprar esses itens.";
		mes "Sim... nós, mercadores, sempre temos dinheiro em nossas mentes, entendeu?";
		close;
	case 3:
		mes "[Membro da Guilda Mahnsoo]";
		mes "Para se tornar um mercador, embora apenas vender e receber dinheiro seja nosso trabalho, vocę deve atingir pelo menos o nível de habilidade básica 9.";
		next;
		mes "[Membro da Guilda Mahnsoo]";
		mes "Bem, costumávamos cobrar uma taxa inicial antes. Năo era difícil conseguir esse dinheiro?";
		mes "Havia muitas pessoas que năo conseguiam juntar aquele dinheiro e ficavam chorando.";
		next;
		mes "[Membro da Guilda Mahnsoo]";
		mes "Entăo, recentemente, decidimos permitir que qualquer Aprendiz que queira ser um mercador se torne um.";
		mes "Porque estar vivo ou năo depois, depende da habilidade de cada um.";
		close;
	case 4:
		close;
	}
}
