				mes "Bem, dou-lhe as boas-vindas";
				mes "de volta de Valhalla e";
				mes "desejo-lhe sorte em sua";
				mes "nova jornada de vida.";
				next;
				skill "NV_TRICKDEAD",0,SKILL_PERM;
				jobchange Job_Acolyte_High;
				skill "AL_HOLYLIGHT",1,SKILL_PERM;
				mes "[Padre Mareusis]";
				mes "Agora, aventure-se e procure aqueles que precisam de sua ajuda. Que Deus ilumine seu caminho.";
				close;
			}else{
				mes "[Padre Mareusis]";
				mes "Agora, aventure-se para procurar pessoas que precisam de sua ajuda. Que Deus ilumine seu caminho.";
				close;
			}
		}else{
			mes "[Padre Mareusis]";
			mes "Sinto que vocę suportou uma experięncia de vida passada. Vocę deve ter aprendido muitas coisas antes de entrar em Valhalla.";
			next;
			mes "[Padre Mareusis]";
			mes "No entanto, posso dizer que vocę năo é adequado para ser um Noviço. Por favor, lembre-se de quem vocę era em sua vida passada e encontre seu caminho.";
			close;
		}
	}
	mes "[Padre Mareusis]";
	mes "O que é que vocę procura?";
	next;
	switch(select("Mudar sua classe para Noviço.:Perguntar os requisitos para ser um Noviço.:Sair.")) {
	case 1:
		mes "[Padre Mareusis]";
		if(BaseJob != Job_Novice) {
			if(BaseJob == Job_Acolyte)
				mes "Vocę está se sentindo bem hoje? Posso dizer pelo seu traje que vocę já é um Noviço. Vocę năo está brincando, está?";
			else
				mes "Sinto muito, mas parece que vocę já tem sua própria Classe, năo é?";
			close;
		}else{
			mes "Vocę realmente deseja se tornar um servo de Deus?";
			mes "Vamos ver se vocę está pronto para isso ou năo... Hmm...";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Padre Mareusis]";
				mes "Oh, céus?! Vocę ainda năo concluiu a prática básica?! Vocę tem um longo caminho a percorrer! Volte depois de aumentar seu Nível de Classe!";
				close;
			}
			mes "[Padre Mareusis]";
			mes "Hmm... seu Nível de Classe é suficiente...";
			mes "Bom. Agora eu lhe darei a qualificaçăo para se tornar um Noviço.";
			next;
			mes "[Padre Mareusis]";
			mes "Lembre-se sempre de ser grato a Deus, que cuida de nós o tempo todo. Em meio ao caos e tempos de dificuldade, enfrente suas tribulaçőes com fé inabalável.";
			callfunc "Job_Change",Job_Acolyte;
			getitem 1545,1;	// N_Mace
			next;
			mes "[Padre Mareusis]";
			mes "Por último, quero sinceramente parabenizá-lo por perseverar em seu julgamento de penitęncia.";
			close;
		}
	case 2:
		mes "[Padre Mareusis]";
		mes "Vocę deseja se tornar um Noviço?";
		mes "Entăo, vocę deve cumprir os seguintes requisitos, pensando neles como as práticas dadas por Deus.";
		next;
		mes "[Padre Mareusis]";
		mes "Primeiro, vocę deve alcançar pelo menos o Nível de Classe 9 de Aprendiz e aprender todas as Habilidades Básicas.";
		mes "Esta é a coisa mais básica a fazer, entăo vocę precisa encarar isso como o caminho para treinar a si mesmo.";
		next;
		mes "[Padre Mareusis]";
		mes "Quando vocę achar que cumpriu este requisito, volte a me procurar. Entăo vocę terá uma Classe sagrada na qual poderá espalhar a vontade de Deus.";
		close;
	case 3:
		close;
	}
}
