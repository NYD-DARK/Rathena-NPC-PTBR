yuno,179,174,4	script	Gomer	851,{
	if (checkweight(1201,1) == 0 || MaxWeight - Weight < 1000) {
		mes "[Gomer]";
		mes "Vocę está carregando muita coisa. Descarte um pouco da sua bagagem e volte para falar comigo.";
		close;
	}
	if (countitem(1549)) {
		mes "[Gomer]";
		mes "Entăo, como o Lança-Estacas está funcionando para vocę?";
		close;
	}
	if (Class != Job_Mechanic && Class != Job_Mechanic_T && Class != Job_Baby_Mechanic) {
		mes "[Gomer]";
		mes "Sou um Mecânico aposentado e năo acho que tenho negócios com vocę.";
		close;
	}
	mes "[Gomer]";
	mes "Entăo? Qual é a sensaçăo de estar em um Madogear?";
	mes "É algo que năo é permitido para aqueles que năo buscaram seu próprio caminho.";
	next;
	mes "[Gomer]";
	mes "Eu? Eu me aposentei há muito tempo. Porque sofri um acidente... fiquei muito ferido e năo serei capaz de pilotar novamente. Estou chateado com isso.";
	next;
	mes "[Gomer]";
	mes "Entăo, decidi ficar aqui forjando engrenagens para Madogear.";
	mes "Hă? Năo estou dizendo que tenho qualquer reclamaçăo sobre este trabalho.";
	next;
	mes "[Gomer]";
	mes "Forçado a fazer algo que vocę năo gosta, năo há nada mais miserável do que isso.";
	next;
	mes "[Gomer]";
	mes "Existem pessoas que săo felizes apenas sendo suportes, sabe, como ser a parede para alguém se encostar.";
	next;
	mes "[Gomer]";
	mes "Isso me faz sentir bem e estou prestes a fazer uma nova engrenagem. Quero que vocę me ajude. O que me diz?";
	next;
	switch(select("I am not that interested.:What are you going to make?:Where do you get the materials?:I brought the materials.")) {
	case 1:
		mes "[Gomer]";
		mes "Năo há nada que eu possa fazer";
		mes "sobre isso entăo.";
		mes "Até a próxima!";
		close;
	case 2:
		mes "[Gomer]";
		mes "Hum, um bastăo de metal enorme que pode destruir um oponente com um golpe!";
		next;
		mes "[Gomer]";
		mes "Năo parece sofisticado, mas projetei um dispositivo com mola para atacar com grande força.";
		next;
		mes "[Gomer]";
		mes "Ele é oculto, entăo quando vocę encontrar uma chance, um disparo pode colocar um oponente em estado crítico!";
		mes "Mal posso esperar para vę-lo em açăo!";
		next;
		mes "[Gomer]";
		mes "Se vocę pudesse imaginar o quăo boa esse tipo de arma pode ser, eu farei isso se tornar realidade.";
		next;
		mes "[Gomer]";
		mes "Traga-me 200 Aços, 30 Tubos Flexíveis e 1 Brocca. Eu poderia fazer o que acabei de descrever com esses materiais.";
		close;
	case 3:
		mes "[Gomer]";
		mes "Vocę deve conseguir aço de muitos monstros, como Kobolds.";
		next;
		mes "[Gomer]";
		mes "Tubos Flexíveis săo mais raros, deixados por Metalings ao redor de Einbroch e Lighthalzen.";
		next;
		mes "[Gomer]";
		mes "Por último, encontrar uma Brocca é muito difícil, mas vocę deve ser capaz de dar um jeito nisso. Haha, vá buscar!";
		close;
	case 4:
		if (countitem(999) < 200 || countitem(7325) < 30 || countitem(1415) < 1) {
			mes "[Gomer]";
			mes "Vocę năo trouxe todos os materiais que pedi.";
			next;
			mes "[Gomer]";
			mes "Eu tenho o hábito de desistir do que estou fazendo e tentar criar coisas novas quando a inspiraçăo surge.";
			mes "É melhor vocę se apressar!";
			close;
		} else {
			mes "[Gomer]";
			mes "Oh, eu estava esperando. Vocę năo veio na hora, entăo eu estava me perguntando se vocę tinha esquecido. Por favor, espere ali um pouco.";
			specialeffect EF_REPAIRWEAPON;
			next;
			mes "[Gomer]";
			mes "Está pronto. O que vocę acha? Nada mal, hein? Embora eu mesmo tenha feito, devo dizer que ficou muito bem feito.";
			next;
			mes "[Gomer]";
			mes "Combinaçăo de peso, volume e veja sua cor e brilho. Isso conquista meu coraçăo!";
			next;
			mes "[Gomer]";
			mes "...........";
			next;
			mes "[Gomer]";
			mes "Uh, com uma pequena melhoria, poderia ser produzido em massa e distribuído. Tenho certeza de que posso fazer mais desses, entăo vou lhe dar este aqui.";
			next;
			mes "[Gomer]";
			mes "Ah, e acontece que eu năo precisei de todo aquele Aço afinal. Entăo aqui, leve a maior parte de volta.";
			delitem 999,50; //Steel
			delitem 7325,30; //Tube
			delitem 1415,1; //Skewer
			getitem 1549,1; //Pilebuncker
			getitem 1360,1; //Two_Handed_Axe
			close;
		}
	}
}
