-	script	Goods Cabinet#00::pggc	-1,{
	mes "É um Armário de Armazenamento de Mercadorias.";
	mes "Uma mensagem está escrita em um pedaço de papel.";
	next;
	mes "+ Cabinet is exclusively +";
	mes "+ for the Eden group +";
	mes "Se você deseja usar este";
	mes "armário, por favor verifique";
	mes "o seguinte:";
	next;
	mes "1.Você é um membro";
	mes "do Grupo do Éden?";
	mes "2.Você aprendeu";
	mes "habilidades básicas o suficiente?";
	mes "3.A taxa do armário é";
	mes "^4d4dff500 zenys^000000!";
	next;
	if (callfunc("F_HasEdenGroupMark")) {
		mes "Você precisa inserir zenys para usar o armário.";
		mes "Custo : 500 Zenys";
		mes "Gostaria de usá-lo?";
		next;
		switch(select("Use the Cabinet.:Cancel.")) {
			case 1:
				if (Zeny > 499) {
					if(!callfunc("F_CanOpenStorage")){
						mes "O armário não está funcionando para mim.";
						mes "Talvez eu ainda não esteja qualificado para usar o Armário.";
						close;
					}
					set Zeny, Zeny - 500;
					close2;
					openstorage;
					end;
				}
				mes "Eu não tenho zenys suficientes.";
				mes "Preciso de pelo menos 500 zenys para usar o Armário.";
				close;
			case 2:
				mes "Vou usar na próxima vez.";
				close;
		}
	}
	mes "Preciso de uma Marca do Grupo do Éden para usar este Armário.";
	close;
}
