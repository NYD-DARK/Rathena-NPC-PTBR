		specialeffect2 EF_BLIND;

		progressbar "ffff00",4;
		delitem .@item_req,.@req_amount;
		if (rand(2)) {
			mes "Vocę usou "+ callfunc("F_InsertPlural",.@req_amount,getitemname(.@item_req)) +" to create ^999900"+ getarg(3) +" Essence^000000.";

			mes "^008800O "+getarg(3)+" Essence will last permanently, unless there's a problem.^000000";
			completequest getarg(0);
			specialeffect2 EF_LORD;
			close;

		mes "^008888Vocę falhou ao refinar a essęncia usando "+ callfunc("F_InsertPlural",.@req_amount,getitemname(.@item_req)) +".^000000";
		close;
	}
	if (checkquest(getarg(0)) == 2) {
		mes "Vocę já criou "+getarg(3)+" Essence in this area.";
		close;
	}
	mes "Alguém já usou esta área.";
	close;
}
