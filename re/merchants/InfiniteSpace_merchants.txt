cmd_fild07,375,167,1	script	Ruins Black Trader#pa0829_01	4_F_JOB_HUNTER,{

Purchase:
	next;
	switch( select( "Give me one", "I don't think so" )) {
		case 1:
			if (!checkweight(22980,1) || (MaxWeight - Weight) < 1000) {
				mes "Você não pode prosseguir com a conversa porque possui uma grande quantidade de itens.";
				mes "Por favor, organize seus itens e tente novamente.";
				close;
			}
			else if (Zeny < 20000) {
				mes "[Mercador Negro das Ruínas]";
				mes "Você não parece ter mais dinheiro. Um pergaminho custa 20.000 Zenys.";
				close;
			}
			Zeny -= 20000;
			getitem 22980,1;
			mes "[Mercador Negro das Ruínas]";
			mes "Foi um bom negócio. Precisa de mais?";
			goto Purchase;
		case 2:
			mes "[Mercador Negro das Ruínas]";
			mes "Volte sempre.";
			close;
	}
}

// Equipment Shop
cmd_fild07,57,275,5	script	Artifact Appraiser#pa0829_01	1_F_02,{
	if (!checkweight(1201,1) || (MaxWeight - Weight) < 1000) {
		mes "Você não pode prosseguir com a conversa porque possui uma grande quantidade de itens.";
		mes "Por favor, organize seus itens e tente novamente.";
		close;
	}
	.@stone_id = 6905;
	mes "[ Avaliador de Artefatos ]";
	mes "Escolha o tipo de equipamento que deseja comprar. Você pode comprar quantos quiser, desde que tenha " + getitemname(.@stone_id) + ".";
	next;
	switch (select("Cancelar:Armas:Armaduras")) {
		case 1:
			mes "[ Avaliador de Artefatos ]";
			mes "Por favor, volte quando quiser~";
			close;

		case 2:
			setarray .@equip_id,1994,1938,13323,13126,28703,2024,16038,21014,28105,18128;
			.@price = 50;
			break;

		case 3:
			setarray .@equip_id,15141,22075,20779,19033;
			.@price = 50;
			break;
	}
	.@menu$ = "Cancel:";
	for (.@i = 0; .@i < getarraysize(.@equip_id); .@i++)
		.@menu$ += getitemname(.@equip_id[.@i]) + ":";
	.@s = select(.@menu$) - 1;
	switch (.@s) {
		case 0:
			mes "[ Avaliador de Artefatos ]";
			mes "Por favor, volte quando quiser~";
			close;

		default:
			.@s--;
			mes "[ Avaliador de Artefatos ]";
			mes "Você precisa de ^0000FF" + .@price + "^000000 " + getitemname(.@stone_id) + " to purchase the " + getitemname(.@equip_id[.@s]) + "~";
			next;
			if (select("Cancelar:Comprar") == 1) {
				mes "[ Avaliador de Artefatos ]";
				mes "Por favor, volte quando quiser~";
				close;
			}
			if (countitem(.@stone_id) < .@price) {
				mes "[ Avaliador de Artefatos ]";
				mes "Você não tem o suficiente de " + getitemname(.@stone_id) + " to purchase this item.";
				close;
			}
			mes "[ Avaliador de Artefatos ]";
			mes "Obrigado pela sua confiança. Por favor, volte novamente na próxima vez.";
			delitem .@stone_id,.@price;
			getitem .@equip_id[.@s],1;
			close;
	}
}

// Equipment Enchanter
cmd_fild07,60,275,3	script	Artifact Enhancer#pa0829_01	4_F_JOB_BLACKSMITH,{
	if (!checkweight(1201,1) || (MaxWeight - Weight) < 1000) {
		mes "Você não pode prosseguir com a conversa porque possui uma grande quantidade de itens.";
		mes "Por favor, organize seus itens e tente novamente.";
		close;
	}
	disable_items;
	.@stone_id = 6905;
	function equip_check;
	mes "[ Aprimorador de Artefatos ]";
	mes "Se você está procurando fortalecer o equipamento que obteve no Espaço Infinito, você veio ao lugar certo.";
	mes "Você tem algum " + getitemname(.@stone_id) + "?";
	next;
	switch (select("Como encanto meu equipamento?:Encantar equipamento.:Inicializar encantamento do equipamento.")) {
		case 1:
			mes "[ Aprimorador de Artefatos ]";
			mes "Você será capaz de obter " + getitemname(.@stone_id) + ", if you explore the space under the Paros Lighthouse.";
			next;
			mes "[ Aprimorador de Artefatos ]";
			mes "Com esse material, posso reforçar o equipamento que você obteve do Espaço Infinito.";
			next;
			mes "[ Aprimorador de Artefatos ]";
			mes "Só posso encantar o 3º e o 4º slot.";
			next;
			mes "[ Aprimorador de Artefatos ]";
			mes "Não há chance do equipamento ser destruído durante o processo de encantamento, mas há uma alta probabilidade de que ele seja destruído ao inicializar os encantamentos.";
			next;
			mes "[ Aprimorador de Artefatos ]";
			mes "Então venha me visitar quando quiser encantar o equipamento.";
			break;

		case 2:
			.@stone_id = 6905;
			.@fee = 20;
			mes "[ Aprimorador de Artefatos ]";
			mes "Por favor, selecione o equipamento que deseja encantar.";
			next;
			switch (select("Cancelar:Arma:Armadura:Calçados:Capa:Elmo")) {
				case 1:
					mes "[ Aprimorador de Artefatos ]";
					mes "Então venha me visitar quando quiser encantar o equipamento.";
					close;

				case 2:
					.@part = EQI_HAND_R;
					break;

				case 3:
					.@part = EQI_ARMOR;
					break;

				case 4:
					.@part = EQI_SHOES;
					break;

				case 5:
					.@part = EQI_GARMENT;
					break;

				case 6:
					.@part = EQI_HEAD_TOP;
					break;
			}
			.@equip_id = getequipid(.@part);
			.@refine = getequiprefinerycnt(.@part);
			equip_check(.@equip_id);
			for (.@i = 0;.@i < 4;.@i++) {
				.@card[.@i] = getequipcardid(.@part,.@i);
				.@check[.@i] = getequipcardid(.@part,.@i);
			}
			if (.@card[2] > 0) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Só posso encantar até o terceiro slot. Seu equipamento não pode ser mais encantado.";
				close;
			}
			switch (.@part) {
				case EQI_HAND_R:
					setarray .@enchant$,
					"4700,4701:4710,4711:4720,4721",
					"4811,4810,4809,4808,4820,4821,4822,4823:4815,4814,4813,4812,4826,4827,4828,4829:4832,4833,4834,4835,4836,4837,4838,4839";
					break;

				case EQI_ARMOR:
				case EQI_SHOES:
					setarray .@enchant$,
					"4700,4701,4702,4703:4710,4711,4712,4713:4720,4721,4722,4723",
					"4795,4796,4797:4870,4871,4800:4870,4871,4800";
					break;

				case EQI_GARMENT:
				case EQI_HEAD_TOP:
					setarray .@enchant$,
					"4700,4701,4702,4703:4710,4711,4712,4713:4720,4721,4722,4723",
					"4861,4862,4867,4868,4900:4861,4862,4867,4868,4900:4861,4862,4867,4868,4900";
					break;
			}
			.@slot = (.@card[3] > 0 ? 2 : 3);
			.@index = (.@slot == 3 ? 0 : 1);
			mes "[ Aprimorador de Artefatos ]";
			mes "Você pode escolher 3 tipos de encantamento, e a taxa de encantamento é " + .@fee + " " + getitemname(.@stone_id) + ".";
			mes "Vou garantir que seja encantado sem problemas.";
			next;
			.@type = select("Sair:Físico:Mágico:Distância") - 2;
			mes "[ Aprimorador de Artefatos ]";
			if (.@slot == 3)
				mes "Ok, vamos começar o primeiro encantamento.";
			else
				mes "Ok, vamos prosseguir com o segundo encantamento.";
			next;
			if (select("Voltarei mais tarde.:Por favor, prossiga.") == 1) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Se mudar de ideia, é só voltar a mim.";
				close;
			}
			if (countitem(.@stone_id) < .@fee) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Hm. Aliás, não acho que você estava ouvindo, você não tem o suficiente de " + getitemname(.@stone_id) + ".";
				close;
			}
			explode(.@T$,.@enchant$[.@index],":");
			explode(.@TT$,.@T$[.@type],",");
			.@enchant = atoi(.@TT$[rand(getarraysize(.@TT$))]);
			.@card[.@slot] = .@enchant;
			delitem .@stone_id,.@fee;
			if (callfunc("F_IsEquipIDHack", .@part, .@equip_id) || callfunc("F_IsEquipRefineHack", .@part, .@refine) || callfunc("F_IsEquipCardHack", .@part, .@check[0], .@check[1], .@check[2], .@check[3]))
				close;
			delequip .@part;
			getitem2 .@equip_id,1,1,.@refine,0,.@card[0],.@card[1],.@card[2],.@card[3];
			specialeffect2 EF_REPAIRWEAPON;
			mes "[ Aprimorador de Artefatos ]";
			mes "Hm. Foi bem feito. Verifique seu equipamento imediatamente.";
			break;
		
		case 3:
			.@stone_id = 6905;
			.@fee = 30;
			.@break_chance = 30;
			mes "[ Aprimorador de Artefatos ]";
			mes "Por favor, selecione o equipamento que deseja encantar.";
			next;
			switch (select("Cancelar:Arma:Armadura:Calçados:Capa:Elmo")) {
				case 1:
					mes "[ Aprimorador de Artefatos ]";
					mes "Então venha me visitar quando quiser encantar o equipamento.";
					close;

				case 2:
					.@part = EQI_HAND_R;
					break;

				case 3:
					.@part = EQI_ARMOR;
					break;

				case 4:
					.@part = EQI_SHOES;
					break;

				case 5:
					.@part = EQI_GARMENT;
					break;

				case 6:
					.@part = EQI_HEAD_TOP;
					break;
			}
			.@equip_id = getequipid(.@part);
			.@refine = getequiprefinerycnt(.@part);
			equip_check(.@equip_id);
			for (.@i = 0; .@i < 4; .@i++) {
				.@card[.@i] = getequipcardid(.@part,.@i);
				.@check[.@i] = .@card[.@i];
			}
			if (.@card[3] == 0) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Seu equipamento não possui nenhum encantamento nele.";
				close;
			}
			mes "[ Aprimorador de Artefatos ]";
			mes "Existe uma chance de seu equipamento do Espaço Infinito ser destruído no processo de inicialização. Você ainda deseja prosseguir?";
			next;
			if (select("Sair:Prosseguir") == 1) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Se mudar de ideia, é só voltar a mim.";
				close;
			}
			if (countitem(.@stone_id) < .@fee) {
				mes "[ Aprimorador de Artefatos ]";
				mes "Hm. Aliás, não acho que você estava ouvindo, você não tem o suficiente de " + getitemname(.@stone_id) + ".";
				close;
			}
			delitem .@stone_id,.@fee;
			if (callfunc("F_IsEquipIDHack", .@part, .@equip_id) || callfunc("F_IsEquipRefineHack", .@part, .@refine) || callfunc("F_IsEquipCardHack", .@part, .@check[0], .@check[1], .@check[2], .@check[3]))
				close;
			delequip .@part;
			if (rand(1,100) > .@chance) {
				getitem2 .@equip_id,1,1,.@refine,0,.@card[0],.@card[1],0,0;
				specialeffect2 EF_REPAIRWEAPON;
				mes "[ Aprimorador de Artefatos ]";
				mes "Hm. Foi bem feito. Verifique seu equipamento imediatamente.";
			} else {
				specialeffect2 EF_REFINEFAIL;
				mes "[ Aprimorador de Artefatos ]";
				mes "Bem, eu avisei. Você não tem sorte, hein?";
			}
			break;
	}
	close;

	function	equip_check	{
		setarray .@equip_id,1994,1938,13323,13126,28703,2024,16038,21014,28105,18128,15141,22075,20779,19033;
		if (!getarg(0)) {
			mes "[ Aprimorador de Artefatos ]";
			mes "Você tirou seu equipamento?";
			close;
		}
		if (inarray(.@equip_id,getarg(0)) == -1) {
			mes "[ Aprimorador de Artefatos ]";
			mes "Este equipamento não é adequado para encantamento. Não se esqueça que apenas a série de equipamentos infinitos pode ser encantada.";
			close;
		}
		return;
	}
