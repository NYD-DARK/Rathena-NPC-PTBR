-	script	::CardSeparation_mal	-1,{
	disable_items;
	if (checkweight(1201,1) == 0) {
		mes "Vocę tem muitos tipos de objetos. Vamos tentar continuar após reduzir esses objetos.";
		close;
	}
	if (MaxWeight - Weight < 10000) {
		mes "Năo é possível continuar porque vocę tem objetos pesados demais. Vamos tentar continuar após reduzir o peso.";
		close;
	}
	if (strnpcinfo(1) == "Jeremy") {
		set .@Jeremy,1;
		set .@n$, "[Jeremy]";
		setarray .@equip_name$[0],  "Armor",  "Shoes",  "Garment", "Upper Hat";
		setarray .@equip_slot[0], EQI_ARMOR,EQI_SHOES,EQI_GARMENT,EQI_HEAD_TOP;

		mes .@n$;
		mes "Quanto tempo năo nos vemos~";
		mes "Eu aprendi uma nova habilidade que separa cartas de Armaduras, Botas, Capas e Equipamentos para a Cabeça. Vocę quer tentar?";
	} else {
		set .@Jeremy,0;
		set .@n$, "[Richard]";
		setarray .@equip_name$[0], "Left hand", "Right hand";
		setarray .@equip_slot[0],   EQI_HAND_L,   EQI_HAND_R;

		mes .@n$;
		mes "O tolo do Jeremy năo quer tocar em armas e escudos por causa de cartas escolhidas que podem ser danificadas. É por isso que preparei uma habilidade de separaçăo de cartas para armas e escudos...";
	}
	next;
	mes .@n$;
	mes "Geralmente a taxa é de 1.000.000 de Zenys. Durante a separaçăo da carta, vocę pode usar ^990000itens especiais que reduzem a taxa de destruiçăo do equipamento ou da carta^000000. Năo cobramos Zenys adicionais por isso.";
	next;
	mes .@n$;
	mes "Existe a possibilidade de destruí-los mesmo usando um item especial. Além disso, ^ff0000o nível de aprimoramento pode ser perdido^000000. Vocę tem algum equipamento para separar?";
	next;

	for(set .@i,0; .@i<getarraysize(.@equip_slot); set .@i,.@i+1) {
		if (getequipisequiped(.@equip_slot[.@i]))
			set .@menu$, .@menu$+getequipname(.@equip_slot[.@i])+":";
		else
			set .@menu$, .@menu$+"^999999"+.@equip_name$[.@i]+" (empty)^000000:";
	}

	set .@i, select("Stop the work:"+((.@Jeremy)?"How is it possible?":"")+":"+.@menu$);
	switch(.@i) {
	case 1:
		mes .@n$;
		mes "Sempre que precisar do serviço, visite-me aqui.";
		close;
	case 2:
		mes .@n$;
		mes "Vocę se pergunta o que há de tăo especial. Bem, eu odeio dar apenas um guia, entăo deixe-me contar a história...";
		next;
		mes "^000099Jeremy está alongando os ombros e as măos. Ele pode estar esperando alguém para conversar com ele.^000000";
		next;
		mes .@n$;
		mes "Vocę sabia que a especialidade de Malangdo é comida enlatada?";
		next;
		select("I knew that well. Is it that limited?");
		mes .@n$;
		mes "Hehe... Todos adoram. Mas houve alguns problemas antes.";
		next;
		select("Problems? Is there any faulty fish?");
		mes .@n$;
		mes "Năo, o peixe năo tem problema nenhum. O problema é a grande quantidade de óleo de peixe produzido após o processamento. Isso é um desperdício industrial.";
		next;
		mes .@n$;
		mes "No entanto, após um processo de refino revitalizante, esse óleo de peixe tornou-se valioso para uso em cuidados com equipamentos antigos e lubrificante industrial.";
		next;
		mes .@n$;
		mes "Além disso, este óleo é tăo útil para separar relíquias de armas equipadas que năo podemos comprar Lubrificante Premium e Lubrificante Comum com Zenys.";
		next;
		mes .@n$;
		mes "Bem, năo se preocupe com dinheiro. Certamente o Lubrificante Premium é caro. Se vocę pagar alguns Zenys, posso lhe dar um lubrificante mais barato.";
		next;
		mes .@n$;
		mes "Năo tenho certeza sobre a qualidade do sucesso. De qualquer forma, isso é bem barato, né?";
		close;
	default:
		set .@equip_num, .@equip_slot[.@i-3];
		if (!getequipisequiped(.@equip_num)) {
			mes .@n$;
			if (.@Jeremy)
				mes "Nesta parte, năo há nada?";
			else
				mes "Năo há nada naquela parte?";
			close;
		}
		break;
	}

	setarray .@check_equip_card[0], getequipcardid(.@equip_num,0),getequipcardid(.@equip_num,1),getequipcardid(.@equip_num,2),getequipcardid(.@equip_num,3);
	setarray .@equip_card[0], .@check_equip_card[0], .@check_equip_card[1], .@check_equip_card[2], .@check_equip_card[3];
	set .@mvp_list$,
		"|4408|4128|4456|4168|4142"+  //Gloom_Under_Night_Card, Golden_Bug_Card, Nidhogg_Shadow_Card, Dark_Lord_Card, Doppelganger_Card
		"|4134|4137|4386|4407|4357"+  //Dracula_Card, Drake_Card, Detale_Card, Randgris_Card, B_Seyren_Card
		"|4146|4132|4147|4372|4145"+  //Maya_Card, Mistress_Card, Baphomet_Card, Bacsojin_Card, Berzebub_Card
		"|4374|4352|4367|4236|4425"+  //Apocalips_H_Card, B_Ygnizem_Card, B_Shecil_Card, Amon_Ra_Card, Atroce_Card
		"|4359|4123|4144|4135|4143"+  //B_Eremes_Card, Eddga_Card, Osiris_Card, Orc_Load_Card, Orc_Hero_Card
		"|4263|4131|4430|4276|4419"+  //Incant_Samurai_Card, Moonlight_Flower_Card, Ifrit_Card, Lord_Of_Death_Card, Ktullanux_Card
		"|4403|4399|4376|4441|4302"+  //Kiel_Card, Thanatos_Card, Lady_Tanee_Card, Fallen_Bishop_Card, Tao_Gunka_Card
		"|4305|4148|4318|4121|4365"+  //Turtle_General_Card, Pharaoh_Card, Knight_Windstorm_Card, Phreeoni_Card, B_Katrinn_Card
		"|4363|4324|4361|4330|4342|"; //B_Magaleta_Card, Garm_Card, B_Harword_Card, Dark_Snake_Lord_Card, Rsx_0806_Card

	if (.@Jeremy) {
		for ( .@i = 0; .@i < MAX_SLOTS; .@i++ ) {
			if (getiteminfo(.@equip_card[.@i], ITEMINFO_SUBTYPE) == CARD_ENCHANT)
				.@equip_card[.@i] = 0;// Armor Enchant System
		}
		if (!getarraysize(.@equip_card)) {
			mes .@n$;
			mes "A carta năo está equipada. Vocę quer verificar novamente?";
			close;
		}
		if ((.@equip_card[0] && compare(.@mvp_list$,"|"+.@equip_card[0]+"|")) ||
			(.@equip_card[1] && compare(.@mvp_list$,"|"+.@equip_card[1]+"|")) ||
			(.@equip_card[2] && compare(.@mvp_list$,"|"+.@equip_card[2]+"|")) ||
			(.@equip_card[3] && compare(.@mvp_list$,"|"+.@equip_card[3]+"|")))
			set .@boss_chk,1;
	} else {
		// Official "Richard" script uses a hardcoded list including every possible item.
		//if (!getequipisequiped(.@equip_num)) {
		//	mes "[Richard]";
		//	mes "Sinto muito. Ainda năo fornecemos esse equipamento.";
		//	close;
		//}

		mes "[Richard]";
		mes "Em qual número de soquete vocę quer separar a carta? A partir do soquete esquerdo, eles săo ordenados como 1, 2, 3, 4.";
		next;
		set .@menu$,"";
		for ( .@i = 0; .@i < MAX_SLOTS; .@i++ ) {
			if (.@equip_card[.@i] && getiteminfo(.@equip_card[.@i], ITEMINFO_SUBTYPE) == CARD_NORMAL) // Armor Enchant System
				.@menu$ = .@menu$ + "Socket " + (.@i+1) + " - " + getitemname(.@equip_card[.@i])+":";
			else {
				.@menu$ = .@menu$ + "^777777Socket " + (.@i+1) + " - No card^000000:";
			}
		}
		set .@i, select("Stop the work:"+.@menu$);
		switch(.@i) {
		case 1:
			mes .@n$;
			mes "Sempre que precisar do serviço, por favor, venha até mim.";
			close;
		default:
			set .@slot, .@i-2;
			if (.@equip_card[.@slot] == 0 || getiteminfo(.@equip_card[.@slot], ITEMINFO_SUBTYPE) == CARD_ENCHANT) {
				mes .@n$;
				mes "Este soquete năo está equipado com nenhuma carta. Por que vocę năo verifica novamente?";
				close;
			}
			break;
		}
		if (compare(.@mvp_list$,"|"+.@equip_card[.@slot]+"|"))
			set .@boss_chk,1;
	}
	if (.@boss_chk == 0) {
		mes .@n$;
		if (.@Jeremy)
			mes "Exceto pelas cartas, ^ff0000todos os efeitos de encantamento desaparecerăo.^000000 Se vocę concorda com isso, por favor, escolha o tipo de trabalho:";
		else
			mes "Por favor, escolha a taxa de serviço.";
		next;
		set .@menu$,
			"Next time...:"+
			((Zeny >= 1000000)?"Use 1,000,000z (Do not use special item):":"^999999Use 1,000,000z (Insufficient)^000000:")+
			((countitem(6441))?"Use Premium Lubricant:":"^999999Premium Lubricant (Insufficient)^000000:")+
			((countitem(6440))?"Use Ordinary Lubricant":"^999999Ordinary Lubricant (Insufficient)^000000");
		switch(select(.@menu$)) {
		case 1:
			mes .@n$;
			mes "Sempre que precisar do serviço, visite-me aqui.";
			close;
		case 2:
			if (Zeny < 1000000) {
				mes .@n$;
				mes "Vocę năo tem Zenys suficientes. Por favor, volte com taxas suficientes.";
				close;
			}
			mes .@n$;
			mes "Este é um equipamento muito antigo. Há uma alta taxa de destruiçăo das cartas ou do equipamento durante o trabalho. Vocę tem certeza de que deseja continuar?";
			next;
			if(select("Da próxima vez...:Continuar") == 1) {
				mes .@n$;
				mes "Sempre que precisar do serviço, visite-me aqui.";
				close;
			}
			set .@sf_c_num,150;
			set .@sf_r_num,150;
			set .@sf_w_num,150;
			set Zeny, Zeny - 1000000;
			break;
		case 3:
			if (countitem(6441) == 0) {
				mes .@n$;
				mes "Vocę năo tem Lubrificante Premium.";
				close;
			}
			mes .@n$;
			mes "Se vocę usar o Lubrificante Premium, a taxa de destruiçăo será reduzida consideravelmente, mas năo posso lhe dar 100% de garantia. Vocę tem certeza de que deseja continuar?";
			next;
			if(select("Da próxima vez...:Continuar") == 1) {
				mes .@n$;
				mes "Sempre que precisar do serviço, visite-me aqui.";
				close;
			}
			set .@sf_c_num,75;
			set .@sf_r_num,75;
			set .@sf_w_num,75;
			delitem 6441,1; //High_RankLubricant
			break;
		case 4:
			if (countitem(6440) == 0) {
				mes .@n$;
				mes "Vocę năo tem Lubrificante Comum.";
				close;
			}
			mes .@n$;
			mes "Se vocę usar o Lubrificante Comum, a taxa de destruiçăo será reduzida consideravelmente, mas năo posso lhe dar 100% de garantia. Vocę tem certeza de que deseja continuar?";
			next;
			if(select("Da próxima vez...:Continuar") == 1) {
				mes .@n$;
				mes "Sempre que precisar do serviço, visite-me aqui.";
				close;
			}
			set .@sf_c_num,75;
			set .@sf_r_num,150;
			set .@sf_w_num,150;
			delitem 6440,1; //General_Lubricant
			break;
		}
	} else if (.@boss_chk == 1) {
		mes .@n$;
		mes "Este equipamento contém uma carta MVP preciosa. Esta carta năo pode ser separada com lubrificante. Se vocę trouxer o super surfactante ^0000ffSillit Pong^000000, eu poderei trabalhar.";
		next;
		switch(select("Da próxima vez...:Tenho um Sillit Pong.")) {
		case 1:
			mes .@n$;
			mes "Sempre que precisar do serviço, visite-me aqui.";
			close;
		case 2:
			if (countitem(6443) == 0) {
				mes .@n$;
				mes "Vocę năo tem Sillit Pong.";
				close;
			}
			break;
		}
		mes .@n$;
		if (.@Jeremy) {
			mes "Exceto pelas cartas, ^ff0000todos os efeitos de encantamento desaparecerăo.^000000 Se vocę concorda com isso, por favor, escolha o tipo de trabalho:";
			set .@menu$,"Alright, let's do it!";
		} else {
			mes "Posso continuar?";
			set .@menu$,"I got it. Just do it quickly!";
		}
		next;
		switch(select("Next time...:"+.@menu$)) {
		case 1:
			mes .@n$;
			mes "Sempre que precisar do serviço, visite-me aqui.";
			close;
		case 2:
			set .@sf_c_num,60;
			set .@sf_r_num,60;
			set .@sf_w_num,60;
			delitem 6443,1; //Sillit_Pong_Bottle
			break;
		}
	}

	set .@equip_id, getequipid(.@equip_num);
	set .@equip_refine, getequiprefinerycnt(.@equip_num);

	// anti-hack
	if (callfunc("F_IsEquipCardHack", .@equip_num, .@check_equip_card[0], .@check_equip_card[1], .@check_equip_card[2], .@check_equip_card[3]))
		close;

	delequip .@equip_num;

	// Chance of retaining refine level.
	if (rand(1,.@sf_r_num) >= 61)
		set .@equip_refine,0;

	if (.@Jeremy) {
		// Chance of retaining equipment.
		if (rand(1,.@sf_w_num) < 61) {
			set .@equip_safe,1;
			getitem2 .@equip_id,1,1,.@equip_refine,0,0,0,0,0;
		}

		// Chance of retaining cards.
		for(set .@i,0; .@i<4; set .@i,.@i+1) {
			if (.@equip_card[.@i]) {
				if (rand(1,.@sf_c_num) < 61)
					getitem .@equip_card[.@i],1;
				else
					set .@card_break,1;
			}
		}
	} else {
		set .@card, .@equip_card[.@slot];
		set .@check_equip_card[.@slot],0;

		// Chance of retaining equipment (all enchantments are preserved).
		if (rand(1,.@sf_w_num) < 61) {
			set .@equip_safe,1;
			getitem2 .@equip_id,1,1,.@equip_refine,0,.@check_equip_card[0],.@check_equip_card[1],.@check_equip_card[2],.@check_equip_card[3];
		}

		// Chance of retaining card.
		if (rand(1,.@sf_c_num) < 61)
			getitem .@card,1;
		else
			set .@card_break,1;
	}

	// Display corresponding effect.
	if (!.@equip_safe && .@card_break)
		specialeffect2 EF_LORD;
	else if (.@equip_safe && .@card_break)
		specialeffect2 EF_SUI_EXPLOSION;
	else if (!.@equip_safe && !.@card_break)
		specialeffect2 EF_FIREPILLAR;
	else
		specialeffect2 EF_MAXPOWER;

	// Output results.
	mes "-- Resultado da Separaçăo de Carta --";
	if (.@equip_safe) {
		mes "Nenhuma rachadura ocorreu durante o processo de separaçăo da carta.";
		mes "^0000FFSeparaçăo do equipamento está normal.^000000";
	} else {
		mes "Ocorreu uma rachadura durante o processo de separaçăo da carta.";
		mes "O equipamento foi danificado. ^ff0000Irrecuperável.^000000";
	}
	mes "-------------------";
	if (!.@card_break) {
		mes "Nenhuma erosăo da superfície ocorreu durante o processo de separaçăo da carta.";
		mes "^0000ffA separaçăo da carta foi bem-sucedida.^000000";
	} else {
		mes "A erosăo da superfície ocorreu durante o processo de separaçăo da carta.";
		mes "A carta foi danificada. ^ff0000Irrecuperável.^000000";
	}
	next;
	mes .@n$;
	mes "Isso é tudo sobre os resultados da separaçăo da carta. Por favor, volte novamente.";
	close;
}
