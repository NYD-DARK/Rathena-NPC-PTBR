job3_guil01,91,93,3	script	Rare Herb Collector	1_M_03,{
	if (checkweight(1201,1) == 0 || MaxWeight - Weight < 2000) {
		mes "- Espere um minuto !! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais com vocę. -";
		mes "- Por favor, tente novamente -";
		mes "- depois que perder um pouco de peso. -";
		close;
	}
	if (Class == Job_Guillotine_Cross || Class == Job_Guillotine_Cross_T || Class == Job_Baby_Guillotine_Cross) {
		mes "[Coletor de Ervas Venenosas Raras]";
		mes "Eu vago pelo mundo coletando ervas venenosas raras. Recentemente, comecei a negociar a erva chamada Izidor. Se estiver interessado, vocę pode comprá-las.";
		next;
		.@i = select("Como posso comprá-las?:Trocar por Sangue de Animal:Trocar por uma Erva Amarga:Trocar por uma Erva Venenosa Mortal:Trocar por uma Rosa Congelada:Trocar por Ment:Trocar por Hinalle")-2;
		if (.@i == -1) {
			mes "[Coletor de Ervas Venenosas Raras]";
			mes "Vocę pode comprá-las por 10.000 zenys e um destes itens raros: Sangue de Animal, Erva Amarga, Erva Venenosa Mortal, Rosa Congelada, Ment ou uma Hinalle.";
			next;
			mes "[Coletor de Ervas Venenosas Raras]";
			mes "Por que peço dinheiro? É apenas para mostrar um pouco de respeito por um coletor como eu. Há, há, há...";
			close;
		} else {
			setarray .@exchange[0],702,621,631,749,605,703;
			.@item = .@exchange[.@i];
			.@price = 10000;

			mes "[Coletor de Ervas Venenosas Raras]";
			mes "Quantas vocę quer?";
			mes "Vocę só pode comprar no máximo 2.000 unidades.";
			mes "Digite '0' se quiser cancelar.";
			next;
			input .@amount;
			if (.@amount == 0) {
				mes "[Coletor de Ervas Venenosas Raras]";
				mes "Vocę cancelou a negociaçăo.";
				close;
			}
			if (.@amount > 2000) {
				mes "[Coletor de Ervas Venenosas Raras]";
				mes "O número deve ser menor que 2.000!";
				close;
			}
			.@total = .@amount * .@price;
			if (countitem(.@item) < .@amount || Zeny < .@total) {
				mes "[Coletor de Ervas Venenosas Raras]";
				mes "Ei, volte quando tiver todos os requisitos para a troca.";
				close;
			}
			mes "[Coletor de Ervas Venenosas Raras]";
			mes "Ótimo. Recebi o dinheiro e o item especial.";
			delitem .@item, .@amount;
			Zeny = Zeny - .@total;
			getitem 709, .@amount; //Izidor
			close;
		}
	}
	mes "[Coletor de Ervas Venenosas Raras]";
	mes "Eu vago pelo mundo coletando ervas venenosas raras. Mas năo sinto vontade de vender minhas ervas para uma pessoa como vocę... há, há, há...";
	close;
}
