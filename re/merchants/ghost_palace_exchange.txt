dali02,51,130,4	script	Dimension Traveler	4_F_SHABBY,{
	if (checkweight(28100,1) == 0) {// Thanos Axe
		mes "- Vocę năo pode prosseguir porque tem muitos itens com vocę. -";
		close;
	}
	.@item_name_req$ = getitemname(6672);// Gray_Shard
	mes "[Viajante Dimensional]";
	mes "Amigo aventureiro, " + mesitemlink( 6672 ) + " is what I need. How about exchanging it with something I have?";
	next;
	.@type = select( "Show me your weapon.", "Show me your armor.", "Anything special?", "I do not need." ) - 1;
	mes "[Viajante Dimensional]";
	switch(.@type) {
	case 0:
		mes "Oh! Vocę está interessado em armas? Nós temos todas as armas que conseguimos do Palácio Fantasma. Por que vocę năo escolhe uma?";
		break;
	case 1:
		mes "Armaduras? Nós temos todas as armaduras que conseguimos do Palácio Fantasma. Por que vocę năo escolhe uma?";
		break;
	case 2:
		mes "Humm! Parece que muitos já ouviram falar que eu tenho as melhores coisas. Sinta-se ŕ vontade para conferir minhas coleçőes.";
		break;
	case 3:
		mes "Adeus, amigo~ Volte sempre!";
		close;
	}
	next;
	mes "[Viajante Dimensional]";
	mes "Ah! O número que está escrito ao lado de " + mesitemlink( 6672 ) + " will be needed.";
	mes "Por favor, tenha isso em mente.";

	switch(.@type) {
	case 0:
		setarray .@items[0],
			13441, 200,		// Thanos Sword
			21009, 200, 	// Thanos Great Sword
			1438,  200,		// Thanos Spear
			1496,  200,		// Thanos Long Spear
			1669,  200,		// Thanos Staff
			2023,  200,		// Thanos Rod
			18119, 200,		// Thanos Bow
			13093, 200,		// Thanos Dagger
			28000, 200,		// Thanos Katar
			1836,  200,		// Thanos Knuckle
			16028, 200,		// Thanos Hammer
			28100, 200,		// Thanos Axe
			1933,  200,		// Thanos Violin
			1988,  200;		// Thanos Whip
		break;
	case 1:
		setarray .@items[0],
			2187,  100,		// Shield Of Gray
			15090, 100, 	// Armor Of Gray
			15091, 100,		// Gray Robe
			20721, 100,		// Cloak Of Gray
			22033, 100,		// Boots Of Gray
			18820, 100;		// Gray Helmet
		break;
	case 2:
		setarray .@items[0],
			27184, 10000,		// Knight Sakray Card
			23676, 50, 			// Gray Charcoal Melee (Melee)
			23677, 50,			// Gray Charcoal Magic (Magic)
			23678, 50;			// Gray Charcoal Range (Range)
		break;
	}
	.@size = getarraysize(.@items);
	for ( .@i = 0; .@i < .@size; .@i += 2 )
		.@menu$ += getitemname(.@items[.@i]) + " - " + .@items[.@i+1] + ":";

	while(true) {
		next;
		.@s = (select(.@menu$) - 1) * 2;
		mes "[Viajante Dimensional]";
		if (.@type != 1)	// armor type doesn't display item info
			mes "Vocę quer dizer " + mesitemlink( .@items[.@s] ) + "?";
		mes "" + .@items[.@s+1] + " ^006400" + .@item_name_req$ + "^000000 are required in exchange.";
		next;
		switch( select( "Let's exchange.", "Let me see other stuff.", "I will come back later." ) ) {
		case 1:
			break;
		case 2:
			mes "[Viajante Dimensional]";
			switch(.@type) {
			case 0:
				mes "Eu tenho armas mais do que suficientes.";
				break;
			case 1:
				mes "Nós temos muitas armaduras para vocę. Por que vocę năo escolhe uma?";
				break;
			case 2:
				mes "Se vocę me trouxer " + .@item_name_req$ + ", I will trade it with anything you want.";
				break;
			}
			continue;
		case 3:
			mes "[Viajante Dimensional]";
			mes "Cuide-se, amigo~ Nos vemos na próxima!";
			close;
		}
		if (countitem(6672) < .@items[.@s+1]) {
			mes "[Viajante Dimensional]";
			mes "Vocę me colocou em uma situaçăo difícil, amigo.";
			mes "Eu preciso de " + .@item_name_req$ + ". I will exchange it for ^006400" + .@items[.@s+1] + " " + .@item_name_req$ + "^000000.";
			close;
		}
		mes "[Viajante Dimensional]";
		mes "Obrigado. Volte se precisar de mais alguma coisa.";
		delitem 6672, .@items[.@s+1];
		getitem .@items[.@s], 1;
		close;
	}
}
