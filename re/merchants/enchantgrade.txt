grademk,34,184,4	script	Sratos#sratos	4_JP_GARM_H,{
	mes "[Sratos]";
	mes "Olá, caro cliente que entrou, vamos torcer por um milagre hoje.";
	mes "Como posso ajudá-lo?";
	next;
	switch( select( "Enhance the equipment's grade.", "Exchange Etel items", "What is Grade Enhancement?" ) ){
		case 1:
			mes "[Sratos]";
			mes "Nosso cliente. Você deseja melhorar o nível (grade) de um equipamento.";
			mes "Não é fácil mover o poder mágico das joias.";
			next;
			mes "[Sratos]";
			mes "O poder mágico pode explodir no local.";
			mes "Então, a arma que herdará o poder mágico também explode!";
			next;
			switch( select( "I'll still do it!", "I'll think about it." ) ){
				case 1:
					mes "[Sratos]";
					mes "Desejo boa sorte aos nossos corajosos clientes!";
					close2;
					enchantgradeui();
					end;
				case 2:
					mes "[Sratos]";
					mes "Sempre que tiver coragem para desafiar, por favor, volte.";
					close;
			}
		case 2:
			mes "[Sratos]";
			mes "Estas são joias usadas para aumentar o nível (grade).";
			mes "Pó de Etel, as joias que servirão de base e, se você me pagar uma pequena taxa, eu trocarei por joias de Etel.";
			close2;
			callshop "EnchantGradeExchange";
			end;

		case 3:
			mes "[Sratos]";
			mes "Ocasionalmente, joias instáveis com muito poder mágico puro são encontradas.";
			mes "Nós chamamos essas de gemas etéricas, certo?";
			next;
			mes "[Sratos]";
			mes "Você pode transferir o poder mágico da gema etérica para outras armas, mas, claro, há um risco.";
			mes "É delicado, então, se não tiver cuidado, pop! Ela explodirá.";
			next;
			mes "[Sratos]";
			mes "De qualquer forma, se você usar esta gema etérica, você pode melhorar a arma que está usando.";
			mes "O desempenho melhora. Chamamos isso de aprimoramento de nível (grade).";
			next;
			mes "[Sratos]";
			mes "Você também, não ficaria feliz se as armas que ama evoluíssem um pouco mais?";
			mes "Nós somos as pessoas que auxiliam nisso. É um pouco arriscado, porém. ahahaha.";
			close;
	}

OnInit:
	setunittitle(getnpcid(0), "<Grade Enhancer>");
	end;
}
