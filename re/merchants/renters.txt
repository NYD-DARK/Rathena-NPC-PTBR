job3_rune01,88,62,5	script	Dragon Breeder	105,{
	mes "[Criador de Dragőes]";
	if ((eaclass()&EAJ_THIRDMASK)==EAJ_RUNE_KNIGHT) {
		mes "Bem-vindo. Vocę gostaria de alugar um Dragăo?";
		next;
		switch(select("Sim:Năo")) {
		case 1:
			//if (!getskilllv("RK_DRAGONTRAINING")) {
			if (!getskilllv("KN_RIDING")) {
				mes "[Criador de Dragőes]";
				mes "Por favor, aprenda a montar em um Dragăo primeiro.";
				close;
			}
			else if (checkdragon()) {
				mes "[Criador de Dragőes]";
				mes "Vocę já tem um Dragăo.";
				close;
			}
			else if (ismounting()) {
				mes "[Criador de Dragőes]";
				mes "Por favor, remova sua montaria de rops.";
				close;
			}
			setdragon;
			close;
		case 2:
			mes "[Criador de Dragőes]";
			mes "Entendo. Entăo, tenha um ótimo dia.";
			close;
		}
	}
	mes "O que vocę está fazendo aqui?";
	mes "Apenas Cavaleiros Rúnicos podem alugar um Dragăo.";
	close;
}

// Mado Gear Renter :: madogear
//============================================================
-	script	::mgm	-1,{
	mes "[Armeiro de Mado Gear]";
	if ((eaclass()&EAJ_THIRDMASK)==EAJ_MECHANIC) {
		mes "Bem-vindo, Mecânico.";
		mes "Vocę gostaria de alugar um Carrinho ou";
		mes "montar um Mado Gear?";
		next;
		switch(select("Alugar um Carrinho:Montar um Mado Gear:Comprar Mado Gear de Emergęncia:Aprimorar Dispositivo de Resfriamento:Cancelar")) {
		case 1:
			if (checkcart()) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę já";
				mes "tem um Carrinho.";
				close;
			}
			setcart;
			mes "[Armeiro de Mado Gear]";
			mes "Aqui está!";
			close;
		case 2:
			if (checkmadogear()) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę já está";
				mes "montado em um Mado Gear.";
				close;
			}
			else if (!getskilllv("NC_MADOLICENCE")) {
				mes "[Armeiro de Mado Gear]";
				mes "Por favor, aprenda a habilidade para obter a Licença de Mado Gear primeiro.";
				close;
			}
			else if(ismounting()) {
				mes "[Armeiro de Mado Gear]";
				mes "Por favor, remova sua montaria de rops.";
				close;
			}
			setmadogear;
			mes "[Armeiro de Mado Gear]";
			mes "Divirta-se e volte sempre!";
			close;
		case 3:
			mes "[Armeiro de Mado Gear]";
			mes "O Mado Gear de Emergęncia é muito útil para situaçőes de emergęncia e é vendido por 1.000.000 de Zenys.";
			next;
			if (select("Comprar:Cancelar") == 2) {
				mes "[Armeiro de Mado Gear]";
				mes "Entendo. Por favor, sinta-se ŕ vontade para me procurar";
				mes "se mudar de ideia.";
				close;
			}
			if (countitem(23277) > 0) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę já possui um Mado Gear de Emergęncia.";
				close;
			}
			if (Zeny < 1000000) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę năo tem Zenys suficientes para comprar o Mado Gear de Emergęncia.";
				close;
			}
			Zeny -= 1000000;
			getitem 23277,1; //Mado_Box
			mes "[Armeiro de Mado Gear]";
			mes "Aqui está!";
			close;
		case 4:
			mes "[Armeiro de Mado Gear]";
			mes "Qual dispositivo vocę deseja aprimorar?";
			next;
			if (select("Dispositivo de Resfriamento:Refrigerador de Alta Qualidade") == 1) {
				mes "[Armeiro de Mado Gear]";
				mes "Aprimorar o Dispositivo de Resfriamento para Refrigerador de Alta Qualidade requer 1 Dispositivo de Resfriamento e 2.000.000 de Zenys.";
				next;
				.@itemid = 2804;
				.@cost = 2000000;
			} else {
				mes "[Armeiro de Mado Gear]";
				mes "Aprimorar o Refrigerador de Alta Qualidade para Refrigerador Especial requer 1 Refrigerador de Alta Qualidade e 4.000.000 de Zenys.";
				next;
				.@itemid = 2809;
				.@cost = 4000000;
			}
			if (select("Aprimorar:Cancelar") == 2) {
				mes "[Armeiro de Mado Gear]";
				mes "Entendo. Por favor, sinta-se ŕ vontade para me procurar";
				mes "se mudar de ideia.";
				close;
			}
			if (!countitem(.@itemid)) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę năo tem o " + getitemname(.@itemid) + ".";
				close;
			}
			if (Zeny < .@cost) {
				mes "[Armeiro de Mado Gear]";
				mes "Sinto muito, mas vocę năo tem Zenys suficientes para aprimorar o dispositivo.";
				close;
			}
			Zeny -= .@cost;
			delitem .@itemid,1;
			getitem (.@itemid == 2804 ? 2809 : 2810),1; //High_Quality_Cooler,Special_Cooler
			mes "[Armeiro de Mado Gear]";
			mes "Aqui está! O seu próprio " + getitemname(.@itemid) + ".";
			close;
		case 5:
			close;
		}
	}
	mes "Como posso ajudar?";
	mes "Mado Gears só estăo disponíveis para Mecânicos.";
	close;
}
