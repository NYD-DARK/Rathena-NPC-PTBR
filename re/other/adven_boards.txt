function	script	F_AvenBoard	{
	.@mo_board = select( "Look up monsters by floor",
					"Monster Information Retrieval System",
					( countitem(909) == 30000 ? "Request" : "" ),
					"Cancel" );
	if (.@mo_board == 1)
		return 1;
	else if (.@mo_board == 2) {
		while(1) {
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			mes "Este é o Sistema de Recuperaçăo de Informaçőes de Monstros. Informaçőes dos monstros que aparecem nesta masmorra podem ser pesquisadas. Insira o nome do monstro, incluindo espaços, ao pesquisar.";
			next;
			input .@inputstr$;
			setarray .@element$[0],"Neutral","Water","Earth","Fire","Wind","Poison","Holy","Shadow","Ghost","Undead";
			setarray .@size$[0],"Medium","Small","Big";
			setarray .@race$[0],"Formless","Undead","Animal","Plant","Insect","Fish","Demon","Demihuman","Angel","Dragon";
			.@size = getargcount();
			.@i = 0;
			while( .@i < .@size && getarg(.@i+1,"") != .@inputstr$ ) .@i += 3;
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			mes "Pesquisando agora";
			next;
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			mes "A pesquisa foi concluída.";
			mes "Exibindo os resultados da pesquisa agora.";
			next;
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			if (.@i >= .@size) {
				mes "Năo há informaçőes disponíveis para esse monstro.";
				mes "Por favor, verifique o nome do monstro.";
			}
			else {
				.@mob_id = getarg(.@i);
				mes "- Nome : "+ getarg(.@i+1);
				mes "- Elemento : "+ .@element$[ getmonsterinfo(.@mob_id,MOB_ELEMENT) ];
				mes "- Tamanho : "+ .@size$[ getmonsterinfo(.@mob_id,MOB_SIZE) ];
				mes "- Raça : "+ .@race$[ getmonsterinfo(.@mob_id,MOB_RACE) ];
				mes "--------------------------------^FD8D02";
				mes getarg(.@i+2);
			}
			next;
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			mes "Gostaria de pesquisar outras informaçőes de monstros?";
			next;
			if (select( "Search","Cancel" ) == 1)
				continue;
			mes "[Sistema de Recuperaçăo de Informaçőes de Monstros]";
			mes "O sistema de pesquisa de monstros será encerrado agora.";
			next;
			return 2;
		}
	}
	else if (.@mo_board == 3) {
		mes "As informaçőes solicitadas aparecerăo - Năo há problema se aparecerem";
		next;
		return 3;
	}
	else if (.@mo_board == 4)
		close;
	else {
		mes "Erro de Operaçăo.";
		close;
	}
}

	mes "Para mais informaçőes sobre os monstros";

	mes "Deve ser utilizado.";
	next;
	return;
}
