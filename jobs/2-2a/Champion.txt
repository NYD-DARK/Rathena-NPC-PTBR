valkyrie,53,42,3	script	Champion#Valkyrie	52,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Mestre]";
			mes "Parabéns.";
			mes "Honra aos guerreiros!";
			close;
		}
		mes "[Mestre]";
		mes "Habilidade.";
		mes "Velocidade.";
		mes "Força.";
		mes "Agilidade.";
		next;
		mes "[Mestre]";
		mes "Um Mestre pode";
		mes "se beneficiar de todas";
		mes "essas coisas. Mas";
		mes "alguém só pode dominar";
		mes "tanto na vida...";
		close;
	}
	else if (ADVJOB == Job_Champion && Class == Job_Acolyte_High && JobLevel > 39) {
		mes "[Mestre]";
		mes "É hora.";
		mes "Hora de grandes heróis";
		mes "se levantarem contra as";
		mes "forças do mal que assolam";
		mes "o mundo de Midgard!";
		next;
		mes "[Mestre]";
		mes "Vocę gostaria";
		mes "de se tornar um Mestre?";
		next;
		if (select("No.:Yes.") == 1) {
			mes "[Mestre]";
			mes "Quando estiver pronto,";
			mes "sinta-se ŕ vontade para voltar.";
			mes "Honra aos guerreiros!";
			close;
		}
		if (SkillPoint) {
			mes "[Mestre]";
			mes "Ainda é possível aprender mais habilidades. Por favor, use";
			mes "todos os seus Pontos de Habilidade restantes antes de voltar a mim.";
			close;
		}
		jobchange Job_Champion;
		set ADVJOB,0;
		mes "[Mestre]";
		mes "Parabéns!";
		mes "Viva como um Mestre,";
		mes "e traga luz ao";
		mes "mundo através da";
		mes "força dos seus punhos.";
		close;
	}
	mes "[Mestre]";
	mes "Bem-vindo";
	mes "a Valhalla,";
	mes "o Salăo da Honra.";
	next;
	mes "[Mestre]";
	mes "Por favor, sinta-se";
	mes "ŕ vontade";
	mes "enquanto estiver aqui.";
	mes "Honra aos guerreiros!";
	close;
}
