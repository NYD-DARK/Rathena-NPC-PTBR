valkyrie,44,50,5	script	MasterSmith#Valkyrie	731,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Mestre Ferreiro]";
			mes "Parabéns.";
			mes "Honra aos guerreiros!";
			close;
		}
		mes "[Mestre Ferreiro]";
		mes "Ah...";
		mes "O ápice";
		mes "do artesanato.";
		mes "Esse é o trabalho de";
		mes "um Mestre Ferreiro.";
		next;
		mes "[Mestre Ferreiro]";
		mes "Uma vez que vocę é";
		mes "o melhor dos melhores,";
		mes "vocę começa a forjar com";
		mes "o objetivo de descobrir";
		mes "maneiras novas e melhores de";
		mes "forjar...";
		close;
	}
	else if (ADVJOB == Job_WhiteSmith && Class == Job_Merchant_High && JobLevel > 39) {
		mes "[Mestre Ferreiro]";
		mes "Chegou a hora!";
		mes "Nosso mundo precisa de aventureiros";
		mes "corajosos e trabalhadores";
		mes "como vocę...";
		next;
		mes "[Mestre Ferreiro]";
		mes "Vocę gostaria de";
		mes "se tornar um Mestre Ferreiro?";
		next;
		if (select("No.:Yes.") == 1) {
			mes "[Mestre Ferreiro]";
			mes "Quando estiver pronto,";
			mes "sinta-se ŕ vontade para voltar.";
			mes "Honra aos guerreiros!";
			close;
		}
		if (SkillPoint) {
			mes "[Mestre Ferreiro]";
			mes "Ainda é possível aprender mais habilidades. Por favor, use";
			mes "todos os seus Pontos de Habilidade restantes antes de retornar a mim.";
			close;
		}
		jobchange Job_Whitesmith;
		set ADVJOB,0;
		mes "[Mestre Ferreiro]";
		mes "Parabéns!";
		mes "Como um Mestre Ferreiro,";
		mes "espero que vocę forje";
		mes "um caminho rumo a um futuro";
		mes "mais brilhante para Midgard.";
		close;
	}
	mes "[Mestre Ferreiro]";
	mes "Bem-vindo";
	mes "a Valhalla,";
	mes "o Salăo da Honra.";
	next;
	mes "[Mestre Ferreiro]";
	mes "Por favor, fique";
	mes "ŕ vontade";
	mes "enquanto estiver aqui.";
	mes "Honra aos guerreiros!";
	close;
}
