aldeba_in,96,181,4	script	Kafra Service	113,{
	cutin "kafra_05",2;
	mes "[Kafra Leilah]";
	mes "Hm...?";
	mes "Oh, bem-vinda ŕ";
	mes "sede da Corporaçăo";
	mes "Kafra. Vocę precisa";
	mes "de algo?";
	next;
	switch(select("Save:Use Storage:Rent a Pushcart:Cancel")) {
	case 1:
		mes "[Kafra Leilah]";
		mes "Seu Ponto de Retorno";
		mes "foi salvo aqui, dentro";
		mes "da sede da Corporaçăo";
		mes "Kafra. Obrigada.";
		next;
		savepoint "aldeba_in",96,179,1,1;
		mes "[Kafra Leilah]";
		mes "Por favor, utilize";
		mes "os serviços Kafra que estăo";
		mes "disponíveis por todo o";
		mes "continente de Midgard. Obrigada por";
		mes "visitar a sede da Kafra.";
		close2;
		break;
	case 2:
		if (Zeny < 20) {
			mes "[Kafra Leilah]";
			mes "Com licença, mas";
			mes "parece que vocę năo";
			mes "tem os 20 zenys para pagar";
			mes "a taxa de acesso ao Armazém...";
			close3;
		}
		mes "[Kafra Leilah]";
		mes "Embora esta instalaçăo seja";
		mes "exclusivamente destinada ao";
		mes "treinamento de Funcionárias Kafra";
		mes "e funçőes administrativas,";
		mes "eu acessarei seu Armazém para vocę.";
		next;
		set Zeny, Zeny-20;
		set RESRVPTS, RESRVPTS + (20/5);
		mes "[Kafra Leilah]";
		mes "No futuro, por favor,";
		mes "peça para a Funcionária Kafra";
		mes "de plantăo se desejar usar";
		mes "qualquer um dos serviços Kafra.";
		mes "Obrigada pela preferęncia.";
		callfunc("F_CheckKafCode");    //check your storage password, if set
		close2;
		openstorage;
		break;
	case 3:
		mes "[Kafra Leilah]";
		mes "Minhas desculpas, mas năo";
		mes "estou de plantăo. Eu ajudaria vocę";
		mes "se pudesse, mas na verdade năo";
		mes "tenho Carrinhos disponíveis.";
		mes "Por que vocę năo pede a outra";
		mes "Funcionária Kafra por assistęncia?";
		next;
		mes "[Kafra Leilah]";
		mes "As Funcionárias Kafra estăo";
		mes "posicionadas por todo o";
		mes "continente de Midgard,";
		mes "e vocę deve conseguir encontrar";
		mes "várias lá fora em Al De Baran.";
		close2;
		break;
	case 4:
		mes "[Kafra Leilah]";
		mes "^666666*Ufa...*^000000";
		mes "Ótimo, porque na verdade";
		mes "estou no meu intervalo";
		mes "agora. Escolher";
		mes "''Cancelar'' foi uma boa";
		mes "atitude da sua parte.";
		close2;
		break;
	}
	cutin "",255;
	end;
}
