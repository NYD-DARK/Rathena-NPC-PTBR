function	script	F_CoolEventCorp	{
	cutin "zonda_01",2;
	mes "[Equipe da Cool Event Corp.]";
	mes "Bem-vindo Ó Cool Event Corp.";
	mes "Nossa equipe estß sempre trabalhando";
	mes "para superar suas expectativas";
	mes "por um serviþo de qualidade. EntÒo, como";
	mes "posso ajudß-lo hoje?";
	next;
	switch(select(getarg(0))) {
	case 1:
		mes "[Equipe da Cool Event Corp.]";
		mes "Seu Ponto de Retorno";
		mes "foi salvo aqui.";
		mes getarg(1)+".";
		mes "Obrigado por usar o";
		mes "serviþo da Cool Event Corp.~";
		savepoint getarg(2),getarg(3),getarg(4),1,1;
		close2;
		break;
	case 2:
		mes "[Equipe da Cool Event Corp.]";
		if(!callfunc("F_CanOpenStorage")){
			mes "Sinto muito, mas vocÛ";
			mes "precisa da habilidade";
			mes "Aperfeiþoar Carrinho nÝvel 6";
			mes "para usar o serviþo de ArmazÚm.";
			close2;
			break;
		}
		if(Zeny < 40){
			mes "Sinto muito, mas vocÛ nÒo";
			mes "tem Zeny suficiente para usar";
			mes "o serviþo de ArmazÚm. Nossa";
			mes "taxa de acesso ao ArmazÚm Ú de 40 Zeny.";
			close2;
			break;
		}
		set Zeny, Zeny - 40;
		mes "Deixe-me abrir seu";
		mes "ArmazÚm pessoal imediatamente.";
		mes "Obrigado por apoiar a Cool";
		mes "Event Corp. ao usar nossos";
		mes "serviþos. Tenha um bom dia~";
		callfunc("F_CheckKafCode");	//check your storage password, if set
		close2;
		openstorage;
		break;
	case 3:
		mes "Por favor, escolha";
		mes "seu destino.";
		next;
		if (VIP_SCRIPT && !vip_status(VIP_STATUS_ACTIVE))
			set .@cost,4400;
		else
			set .@cost,2200;
		.@destination$ = getarg(5);
		if (select(.@destination$ +" -> "+ .@cost +" z:Cancel") == 1) {
			if ((countitem(7060) != 0) || (Zeny >= .@cost)) {
				if (countitem(7060) != 0) {
					delitem 7060,1; //Warp_Free_Ticket
				}
				else {
					set Zeny,Zeny-.@cost;
				}
				if (.@destination$ == "Veins") warp "veins",205,101;
				else if (.@destination$ == "Rachel") warp "rachel",115,125;
				cutin "",255;
				end;
			}
			else {
				mes "[Equipe da Cool Event Corp.]";
				mes "Sinto muito, mas vocÛ nÒo tem";
				mes "zeny suficiente para o serviþo de";
				mes "Teleporte. A taxa para se teletransportar";
				mes "to "+ .@destination$ +" is "+.@cost+" zeny.";
			}
		}
		close2;
		break;
	case 4:
		mes "[Equipe da Cool Event Corp.]";
		if(BaseClass != Job_Merchant){
			mes "Sinto muito, mas o";
			mes "serviþo de aluguel de Carrinho";
			if (checkre(0))
				mes "s¾ estß disponÝvel para a Classe Mercador.";
			else {
				mes "s¾ estß disponÝvel para Mercadores,";
				mes "Ferreiros, Mestres Ferreiros,";
				mes "Alquimistas e Criadores.";
			}
			close2;
			break;
		}
		if(getskilllv("MC_PUSHCART") == 0){
			mes "VocÛ s¾ pode alugar um Carrinho ap¾s";
			mes "aprender a habilidade Carrinho.";
			close2;
			break;
		}
		if(checkcart() == 1){
			mes "VocÛ jß tem";
			mes "um Carrinho equipado.";
			mes "Infelizmente, nÒo podemos";
			mes "alugar mais de um para";
			mes "cada cliente por vez.";
			close2;
			break;
		}
		if(countitem(7061)){
			delitem 7061,1; //Cart_Free_Ticket
			setcart;
			close2;
			break;
		}
		mes "A taxa de aluguel";
		mes "do Carrinho Ú 800 Zeny. VocÛ";
		mes "gostaria de alugar um Carrinho?";
		next;
		if(select("Rent a Pushcart:Cancel") == 1){
			if(Zeny < 800){
				mes "[Equipe da Cool Event Corp.]";
				mes "Sinto muito, mas vocÛ";
				mes "nÒo tem Zeny";
				mes "suficiente para pagar a taxa de";
				mes "aluguel do Carrinho de 800 Zeny.";
			} else {
				set Zeny, Zeny - 800;
				setcart;
			}
		}
		close2;
		break;
	case 5:
		callfunc("F_SetKafCode","[Cool Event Corp. Staff]","Cool Event Corp.");
		//the function closes msg and removes cutins
	case 6:
	default:
		mes "[Equipe da Cool Event Corp.]";
		mes "A Cool Event Corp. estß sempre";
		mes "se esforþando para oferecer os melhores";
		mes "serviþos para nossos clientes.";
		mes "Ajude-nos a ser os melhores";
		mes "nos enviando suas opini§es";
		mes "e feedback honesto. Obrigado.";
		close2;
	}
	cutin "", 255;
	end;
}

		"in the town of Lighthalzen","lighthalzen",95,240;
	end;
