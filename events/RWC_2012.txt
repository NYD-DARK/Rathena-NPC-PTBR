prontera,147,61,3	script	Driller#pron	87,{
	disable_items;
	if (checkweight(1201,1) == 0) {
		mes "Vocę está carregando itens demais em seu inventário. Visite o armazém da Kafra e tente novamente.";
		close;
	}
	if (MaxWeight - Weight < 10000) {
		mes "Vocę năo pode prosseguir porque está sobrecarregado.";
		close;
	}
	set .@part, EQI_ACC_L;
	if (!getequipisequiped(.@part)) {
		mes "[Perfurador]";
		mes "Minha classe é perfurar um slot de carta em acessórios RWC Memorial.";
		next;
		mes "[Perfurador]";
		mes "Sinto muito, mas vocę năo tem nenhum item equipado na posiçăo de acessório direito.";
		close;
	}
	mes "[Perfurador]";
	mes "Minha classe é perfurar um slot de carta em acessórios RWC Memorial. Além disso, eu apenas trato ^ff0000itens puros^000000, ou aqueles que năo foram encantados.";
	next;
	set .@equip_id, getequipid(.@part);
	if (.@equip_id != 2966 && .@equip_id != 2968) {
		mes "[Perfurador]";
		mes "No entanto, vejo que o acessório que vocę está usando no lado direito năo pode ser tratado. Por favor, equipe um acessório RWC Memorial.";
		close;
	}
	mes "[Perfurador]";
	mes "Vocę também deve saber que fazer um slot de carta é extremamente perigoso. ^ff0000As chances de sucesso săo de cerca de 50%.^000000 Deseja prosseguir?";
	next;
	if(select("Cancel:Let's go!") == 1) {
		mes "[Perfurador]";
		mes "Até mais, entăo.";
		close;
	}
	if (.@equip_id == 2966) {
		set .@slotted, 2967; //RWC_2012_Ring_
		set .@name$,"RWC 2012 Memorial Ring";
		set .@str$,"ring";
	} else if (.@equip_id == 2968) {
		set .@slotted, 2969; //RWC_2012_Pendant_
		set .@name$,"RWC 2012 Memorial Pendant";
		set .@str$,"pendant";
	} else {
		mes "[Perfurador]";
		mes "Năo consigo identificar o acessório que vocę está usando na măo direita. Năo posso trabalhar nele.";
		close;
	}
	if (getequipcardid(.@part,3) > 0) {
		mes "[Perfurador]";
		mes "Este item já foi encantado. Năo posso trabalhar nele, pois é contra as regras.";
		close;
	}

	// anti-hack
	if (callfunc("F_IsEquipIDHack", .@part, .@equip_id))
		close;

	delequip .@part;
	if (rand(1,10) > 5) {
		getitem .@slotted,1;
		specialeffect2 EF_REPAIRWEAPON;
		mes "[Perfurador]";
		mes "Yay! Success! Your "+.@name$+" now has a card slot. Check it out!";
		close;
	} else {
		specialeffect2 EF_LORD;
		mes "[Perfurador]";
		mes "Awww... Damn weak "+.@str$+"... It broke during the procedure. I'm sorry.";
		close;
	}
}

prontera,147,59,3	script	Goldberg#pron	878,{
	disable_items;
	if (checkweight(1201,1) == 0) {
		mes "Vocę está carregando itens demais. Volte depois de organizar seu inventário.";
		close;
	}
	if (MaxWeight - Weight < 10000) {
		mes "Vocę năo pode prosseguir porque está sobrecarregado.";
		close;
	}
	mes "[Goldberg]";
	mes "Olá! Sou responsável por encantar acessórios RWC Memorial com alguns poderes místicos.";
	next;
	set .@part, EQI_ACC_L;
	if (!getequipisequiped(.@part)) {
		mes "[Goldberg]";
		mes "Sinto muito, mas vocę năo tem nenhum item equipado na posiçăo de acessório direito.";
		close;
	}
	set .@equip_id, getequipid(.@part);
	if (.@equip_id < 2966 || .@equip_id > 2969) {
		mes "[Goldberg]";
		mes "No entanto, vejo que o acessório que vocę está usando năo é algo em que eu possa trabalhar. Por favor, equipe um acessório RWC Memorial.";
		close;
	}
	set .@select, select("Sorry, not interested.:Please, empower my accessory.:Remove the Enchant.")-1;
	if (.@select == 0) {
		mes "[Goldberg]";
		mes "Tudo bem, entăo, até a próxima...";
		close;
	}
	setarray .@equip_card[0], getequipcardid(.@part,0),getequipcardid(.@part,1),getequipcardid(.@part,2),getequipcardid(.@part,3);
	if (.@select == 1) {
		switch(getequipid(.@part)) {
		case 2966:
			setarray .@option[0],2,2,4,4;
			break;
		case 2967:
			setarray .@option[0],0,2,4,4;
			break;
		case 2968:
			setarray .@option[0],1,1,3,3;
			break;
		case 2969:
			setarray .@option[0],0,1,3,3;
			break;
		default:
			mes "[Goldberg]";
			mes "Sinto muito, mas năo posso trabalhar no acessório que vocę está usando atualmente.";
			mes "Se vocę equipou seu acessório RWC Memorial no lado esquerdo, tente trocá-lo para o lado direito.";
			close;
		}
		for(set .@i,3; .@i>=0; set .@i,.@i-1) {
			if (.@equip_card[.@i] == 0) {
				set .@slot, .@i;
				set .@op_type, .@option[.@i];
				break;
			}
		}
		switch(.@op_type) {
		case 4:
			mes "[Goldberg]";
			mes "Qual encantamento vocę gostaria de infundir?";
			next;
			setarray .@enchant_select[0],1,2,3,4;
			set .@i, select("Cancel:Fighting Spirit:ATK (%):Max HP:HP")-2;
			break;
		case 3:
			setarray .@enchant_select[0],5,6,7;
			set .@i, select("Cancel:Spell:MATK (%):SP")-2;
			break;
		case 2:
			mes "[Goldberg]";
			mes "^ff0000Cuidado! Há cerca de 25% de chance do encantamento falhar. Se isso acontecer, o item será destruído.^000000 Qual encantamento vocę gostaria de infundir?";
			next;
			setarray .@enchant_select[0],8,9,10,11,12,13,14;
			set .@i, select("Cancel:STR:AGI:VIT:INT:DEX:LUK:SP")-2;
			break;
		case 1:
			mes "[Goldberg]";
			mes "^ff0000Há cerca de 25% de chance do encantamento falhar. Se isso acontecer, o item será destruído.^000000 Qual encantamento vocę gostaria de infundir?";
			next;
			setarray .@enchant_select[0],8,9,10,11,12,13,15,16;
			set .@i, select("Cancel:STR:AGI:VIT:INT:DEX:LUK:MHP:HP")-2;
			break;
		case 0:
			mes "[Goldberg]";
			mes "Seu acessório recebeu tantos encantamentos que quase năo consigo mais trabalhar nele.";
			close;
		}
		if (.@i == -1) {
			mes "[Goldberg]";
			mes "Tudo bem, entăo, até a próxima.";
			close;
		}
		mes "[Goldberg]";
		mes "O poder do encantamento será escolhido aleatoriamente. ^ff0000Uma vez infundido, o encantamento năo pode ser removido.^000000 Vamos continuar?";
		next;
		if(select("No, please stop.:Yes, please proceed.") == 1) {
			mes "[Goldberg]";
			mes "Tudo bem, entăo, até a próxima...";
			close;
		}
		set .@enchant_type, .@enchant_select[.@i];
		if (!getequipisequiped(.@part)) {
			mes "[Goldberg]";
			mes "Năo retire seu equipamento enquanto eu estiver trabalhando, ok?";
			close;
		}
		switch(.@enchant_type) {
		case 1:
			setarray .@enc[0],4811,4810,4809; //Fighting_Spirit1,Fighting_Spirit2,Fighting_Spirit3
			break;
		case 2:
			setarray .@enc[0],4819,4766,4767; //Atk1,Atk2,Atk3
			break;
		case 3:
			setarray .@enc[0],4861,4862,4867; //MHP1,MHP2,MHP3
			break;
		case 4:
			setarray .@enc[0],4795,4796,4797; //HP100,HP200,HP300
			break;
		case 5:
			setarray .@enc[0],4760,4761,4806; //Matk1,Matk2,Matk3
			break;
		case 6:
			setarray .@enc[0],4815,4814,4813; //Spell1,Spell2,Spell3
			break;
		case 7:
			setarray .@enc[0],4870,4800,4871; //SP25,SP50,SP75
			break;
		case 8:
			setarray .@enc[0],4700,4701,4702; //Strength1,Strength2,Strength3
			break;
		case 9:
			setarray .@enc[0],4730,4731,4732; //Agility1,Agility2,Agility3
			break;
		case 10:
			setarray .@enc[0],4740,4741,4742; //Vitality1,Vitality2,Vitality3
			break;
		case 11:
			setarray .@enc[0],4710,4711,4712; //Inteligence1,Inteligence2,Inteligence3
			break;
		case 12:
			setarray .@enc[0],4720,4721,4722; //Dexterity1,Dexterity2,Dexterity3
			break;
		case 13:
			setarray .@enc[0],4750,4751,4752; //Luck1,Luck2,Luck3
			break;
		case 14:
			setarray .@enc[0],4870,4800,4871; //SP25,SP50,SP75
			break;
		case 15:
			setarray .@enc[0],4861,4862,4867; //MHP1,MHP2,MHP3
			break;
		case 16:
			setarray .@enc[0],4795,4796,4797; //HP100,HP200,HP300
			break;
		default:
			mes "[Goldberg]";
			mes "Tivemos um problema, deixe-me verificar.";
			close;
		}

		if (.@enchant_type < 8)
			set .@i, rand(1,300); // 0% break chance.
		else
			set .@i, rand(1,400); // 25% break chance.

		     if (.@i < 151) set .@enchant, .@enc[0];
		else if (.@i < 251) set .@enchant, .@enc[1];
		else if (.@i < 301) set .@enchant, .@enc[2];
		else set .@enchant,9;

		// anti-hack
		if (callfunc("F_IsEquipIDHack", .@part, .@equip_id) ||
		    callfunc("F_IsEquipCardHack", .@part, .@equip_card[0], .@equip_card[1], .@equip_card[2], .@equip_card[3]))
			close;

		set .@equip_card[.@slot], .@enchant;
		if (.@slot == 2 && .@enchant == 0) {
			set .@equip_card[3],0;
		} else if (.@slot == 1 && .@enchant == 0) {
			set .@equip_card[2],0;
			set .@equip_card[3],0;
		} else if (.@slot == 0 && .@enchant == 0) {
			set .@equip_card[1],0;
			set .@equip_card[2],0;
			set .@equip_card[3],0;
		}

		set .@equip_refine, getequiprefinerycnt(.@part);
		delequip .@part;
		if (.@enchant == 9) {
			mes "[Goldberg]";
			mes "Oh, céus!";
 			mes "O item năo era forte o suficiente para suportar o encantamento e, portanto, foi destruído. Sinto muito.";
			specialeffect2 EF_LORD;
			close;
		}
		if (.@enchant == 0) { // Should never happen.
			mes "[Goldberg]";
			mes "Oh... Parece que houve uma instabilidade de algum tipo entre todos os poderes infundidos. Isso fez com que todos os encantamentos desaparecessem. É uma pena, mas por favor, tente novamente!";
		} else {
			mes "[Goldberg]";
			mes "Ótimo!";
			mes "The enchantment is a success! It will be applied in socket No.^990000"+(.@slot+1)+"^000000.";
			specialeffect2 EF_REPAIRWEAPON;
		}

//		GetNonSlotItemSock2 .@equip_refine .@equip_id .@equip_card[0] .@equip_card[1] .@equip_card[2] .@equip_card[3]
		getitem2 .@equip_id,1,1,.@equip_refine,0,.@equip_card[0],.@equip_card[1],.@equip_card[2],.@equip_card[3];

		close;
	} else if (.@select == 2) {
		mes "[Goldberg]";
		mes "Vou apenas inicializar a opçăo de encantamento sem fazer nada com a carta equipada no slot. Vocę quer continuar?";
		next;
		if(select("I will stop.:Yep, sure, go on.") == 1) {
			mes "[Goldberg]";
			mes "Volte se mudar de ideia.";
			close;
		}
		if (countitem(6665) == 0) {
			mes "[Goldberg]";
			mes "Sinto muito. Mas vocę năo tem o cupom de Inicializaçăo RWC. Pode verificar seu inventário?";
			close;
		}
		if (getiteminfo(.@equip_card[3], ITEMINFO_SUBTYPE) != CARD_ENCHANT) {
			mes "[Goldberg]";
			mes "Hm... este equipamento está limpo. Năo posso inicializá-lo se năo houver nada! Verifique novamente.";
			close;
		}
		specialeffect2 EF_REPAIRWEAPON;
		mes "[Goldberg]";
		mes "A opçăo de encantamento em seu item será inicializada.";
		delitem 6665,1; //RWC_Inicializer
		delequip .@part;
	
//		GetNonSlotItemSock2 .@equip_refine .@equip_id .@equip_card[0] .@equip_card[1] .@equip_card[2] .@equip_card[3]
		for ( .@i = getiteminfo(.@equip_id, ITEMINFO_SLOT); .@i < MAX_SLOTS; .@i++ ) {
			if (getiteminfo(.@equip_card[.@i], ITEMINFO_SUBTYPE) == CARD_ENCHANT)
				.@equip_card[.@i] = 0;// Armor Enchant System
		}
		getitem2 .@equip_id,1,1,.@equip_refine,0,.@equip_card[0],.@equip_card[1],.@equip_card[2],.@equip_card[3];
		
		close;
	}
}
