function	script	F_GldManager	{
	.@npc_name$ = getarg(0);
	.@map_name$ = getarg(1);
	.@gid = getcastledata(.@map_name$,1);
	.@gm_name$ = getguildmaster(.@gid);

	mes "[ "+ .@npc_name$ +" ]";
	if (.@gid == 0){
		mes "Estou esperando pelo meu mestre. Bravo aventureiro, siga o seu destino!";
		return;
	}
	if (getcharid(2) != .@gid){
		mes "Estou aqui para seguir as ordens de ^5533FF" + getguildmaster(.@gid) + "^000000's command! Hey! Your not even a part of the guild!!";
		mes "Onde estăo os guardiőes? Destruam estes intrusos!";
		return;
	}
	if (is_guild_leader() == false){
		mes "Vocę năo é ^5533FF" + getguildmaster(.@gid) + "^000000! I am here to follow ^5533FF" + getguildmaster(.@gid) + "^000000's command only";
		return 0;
	}

	mes "Bem-vindo, Mestre ^5533FF" + getguildmaster(.@gid) + "^000000 ! I will assist you in any way I can!";
	next;

	menu "Kafra Staff Employment / Dismissal",M_Kaf, "Enter Treasure Room",M_Treas, "Cancel",M_End;

	//===========================
	M_Kaf:
	//======
		mes "[ "+ .@npc_name$ +" ]";
		if (GetCastleData(.@map_name$,9) == 1) goto L_Dismiss;
		if (getgdskilllv(.@gid,10001) == 0){
			mes "Mestre, vocę năo tem um contrato com a Corporaçăo Kafra.";
			mes "Para contratar uma Kafra, vocę deve primeiro aprender a habilidade de Clă ^5533FFContrato com a Kafra^000000.";
			return;
		}

		L_Hire:
			mes "Gostaria de contratar os serviços de uma Kafra? Vocę precisará de ^5533FF10.000 Zeny^000000 para fazer isso... ";
			next;
			menu "Employ Kafra.",-,"Cancel",sM_KafEnd;

				mes "[ "+ .@npc_name$ +" ]";
				if (Zeny < 10000){
					mes "Mestre, vocę năo tem dinheiro suficiente para contratar uma Kafra. A contrataçăo foi cancelada.";
					return;
				}
				set Zeny,Zeny-10000;
				enablenpc "Kafra Staff#"+getarg(4);
				SetCastleData .@map_name$,9,1;
				mes "Vocę firmou um contrato com a Corporaçăo Kafra.";
				next;
				cutin "kafra_01",2;
				mes "[ Funcionária Kafra ]";
				mes "Como vai? Estou aqui para lhe oferecer um serviço útil! Farei o melhor que puder para servi-lo.";
				next;
				cutin "kafra_01",255;
				mes "[ "+ .@npc_name$ +" ]";
				//mes "Seu contrato de trabalho dura ^5533FF1 męs^000000. Após o término deste prazo, vocę precisará criar um novo contrato.";
				mes "Acho que a Funcionária Kafra trará benefícios para os membros do nosso clă.";
				return;

			sM_KafEnd:
				mes "[ "+ .@npc_name$ +" ]";
				mes "Como desejar, Mestre. Mas sugiro que consigamos uma Kafra o quanto antes!";
				return;

		L_Dismiss:
			mes "Gostaria de dispensar a Kafra atual?";
			next;
			menu "Dismissal",-,"Cancel",sM_KafEnd2;

				cutin "kafra_01",2;
				mes "[ Funcionária Kafra ]";
				mes "Eu fiz algo de errado? Se fiz, por favor, vocę me perdoa?";
				next;
				menu "Dismiss",-,"Cancel",ssM_KafEnd2;

					mes "[ Funcionária Kafra ]";
					mes "É uma pena que eu năo poderei mais servir ao seu clă....";
					next;
					disablenpc "Kafra Staff#"+getarg(4);
					SetCastleData .@map_name$,9,0;
					cutin "kafra_01",255;
					mes "[ "+ .@npc_name$ +" ]";
					mes "A Kafra foi dispensada. Mas... nós realmente deveríamos conseguir uma Kafra o quanto antes!";
					return;
				ssM_KafEnd2:
					mes "[ Funcionária Kafra ]";
					mes "Obrigada, mestre, farei o meu melhor! ^^.";
					cutin "kafra_01",255;
					return;
			sM_KafEnd2:
				mes "[ "+ .@npc_name$ +" ]";
				mes "Mestre, acho que vocę deveria manter a atual Funcionária Kafra, pois ela já está se esforçando ao máximo para nos servir";
				return;

	//=========================
	M_Treas:
	//========
		mes "[ "+ .@npc_name$ +" ]";
		mes "Gostaria de ir para a nossa Sala do Tesouro? Apenas vocę, o Líder do Clă, tem permissăo para entrar nesta sala.";
		next;
		menu "Enter Treasure room.",-,"Cancel",sM_TresEnd;

			mes "[ "+ .@npc_name$ +" ]";
			mes "Por favor, siga-me pela passagem secreta.";
			mes "Vocę deve puxar o interruptor secreto para conseguir sair.";
			next;
			warp .@map_name$,getarg(2),getarg(3);
			return;
		sM_TresEnd:
			mes "[ "+ .@npc_name$ +" ]";
			mes "As mercadorias săo produzidas todos os dias.";
			mes "Vocę deve pegá-las sempre que puder, pois elas podem desaparecer se vocę tentar pegá-las no momento errado.";
			return;


	//==========================
	M_End:
	//=======
		mes "[ "+ .@npc_name$ +" ]";
		mes "Como desejar, mestre.";
		return;
}
