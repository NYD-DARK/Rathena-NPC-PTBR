xmas_in,100,96,4	script	Father Christmas::Santa2	718,{
	mes "[Papai Noel]";
	if(xmas_npc==0) set xmas_npc, 1; // For Lutie & Bard quest
	if(#event_xmas > 0 && #event_xmas < 30 ) goto L_Start;
	mes "Feliz Natal!";
	if(Class==0 || #event_xmas>=30 ) close; //anti exploit protection
	mes "Tenho um presente para vocę! Ho Ho Ho!";
	getitem rand(664,667),1; //gives one of 4 gift boxes
	set #event_xmas,#event_xmas+1;
	close;

L_Start:
	mes "Estou com um pequeno problema...";
	mes "Vocę se importa de ouvir?";
	next;
	menu "Listen to Santa Claus.",M_0, "Give Santa Claus proof.",M_1, "Cancel.",M_End;

	M_0:
		mes "[Papai Noel]";
		mes "Meu problema é este.";
		mes "Parece haver um homem por aí";
		mes "que está se passando por mim e espalhando";
		mes "terror por toda a terra.";
		next;
		mes "[Papai Noel]";
		mes "Como o lendário Grinch, ele está pegando";
		mes "todos os brinquedos das crianças e guardando-os";
		mes "para si mesmo.";
		next;
		mes "[Papai Noel]";
		mes "Estou muito ocupado aqui criando meu lote de";
		mes "brinquedos para o próximo ano, entăo năo posso";
		mes "sair e encontrá-lo sozinho.";
		mes "Entăo, eu gostaria que vocę fosse lá e";
		mes "Destruísse esse homem para mim.";
		next;
		mes "[Papai Noel]";
		mes "Ele está em posse de um dos meus";
		mes "sacos mágicos, no entanto, entăo ele";
		mes "fugirá para outro lugar toda vez";
		mes "que vocę o atacar.";
		next;
		mes "[Papai Noel]";
		mes "No entanto,";
		mes "Na pressa, ele tende a deixar cair coisas.";
		mes "Se por acaso ele deixar cair uma de suas Meias";
		mes "Furadas que ele usa para roubar os";
		mes "brinquedos das pobres crianças, pegue-a.";
		next;
		mes "[Papai Noel]";
		mes "Se vocę coletar 3 delas, eu lhe darei";
		mes "uma caixa misteriosa protótipo que";
		mes "eu tenho mantido pelo";
		mes "laboratório. Ela gera presentes aleatórios";
		mes "e me economiza muito trabalho.";
		close;
	M_1:
		mes "[Papai Noel]";
		if(countitem(7034) < 3) goto L_NotEnuf;
		delitem 7034,3;
		mes "Parece que vocę tem feito um";
		mes "bom trabalho derrotando esses";
		mes "Papai Noéis falsos. Continue assim!";
		next;
		getitem 644,1; //Gift Box Prototype
		set #event_xmas,#event_xmas+1;
		mes "[Papai Noel]";
		mes "Aqui está sua recompensa.";
		mes "Se vocę conseguir mais 3, eu lhe darei outra.";
		mes "Espero que vocę consiga um bom item.";
		close;
	
		L_NotEnuf:
			mes "Vocę năo tem meias suficientes como prova.";
			mes "Vá derrotar esses Papai Noéis malignos";
			mes "e consiga mais para mim e eu o recompensarei.";
			close;
	M_End:
		mes "[Papai Noel]";
		mes "Entendo. Bem, pelo menos";
		mes "nos encontraremos novamente na manhă de Natal.";
		close;

OnInit:
	disablenpc "Santa Claus";
	end;
}
