prontera,113,294,3	script	Sign Post#god	837,{
	mes "======== Status do Selo Divino ========";
	setarray .@seal$[1], "Sleipnir","Megingjard","Brisingamen","Mjolnir";
	for(set .@i,1; .@i<=4; set .@i,.@i+1) {
		set .@val, getd("$God"+.@i);
		if (.@val == 0)
			set .@status$,"Unseen";
		else if (.@val < $@god_check1)
			set .@status$,"Active";
		else if (.@val < $@god_check2)
			set .@status$,"Appeared";
		else
			set .@status$,"Released";
		mes .@seal$[.@i]+" Seal: "+.@status$;
	}
	mes " ";
	mes "======= Status do Seu Selo ========";
	if (god_sl_1 < 51)
		mes "^ff0000Selo de Sleipnir: N„o ConcluÌdo^000000";
	else
		mes "^00ff00Selo de Sleipnir: ConcluÌdo^000000";
	if (god_eremes < 28)
		mes "^ff0000Selo de Megingjard: N„o ConcluÌdo^000000";
	else
		mes "^00ff00Selo de Megingjard: ConcluÌdo^000000";
	if (god_brising < 50)
		mes "^ff0000Selo de Brisingamen: N„o ConcluÌdo^000000";
	else
		mes "^00ff00Selo de Brisingamen: ConcluÌdo^000000";
	if (god_mjo_0 < 11)
		mes "^ff0000Selo de Mjolnir: N„o ConcluÌdo^000000";
	else
		mes "^00ff00Selo de Mjolnir: ConcluÌdo^000000";
	close;
}
