payon_in03,188,146,5	script	Marx Hansen#juice	86,{
	if (checkweight(1201,1) == 0) {
		mes "- Espere um momento! -";
		mes "- Atualmente vocę está carregando -";
		mes "- muitos itens com vocę. -";
		mes "- Por favor, volte mais tarde -";
		mes "- depois de colocar alguns itens no armazém da Kafra. -";
		close;
	}
	if (MISC_QUEST&1) {
		mes "[Mercador Marx Hansen]";
		mes "Bem-vindo.";
		mes "Vocę veio para";
		mes "processar frutas também?";
		next;
		switch(select("Make Juice.:Talk and get information about fruit processing.:Cancel.")) {
		case 1:
			mes "[Mercador Marx Hansen]";
			mes "Que tipo de suco de fruta vocę gostaria de fazer?";
			next;
			switch(select("Apple Juice:Banana Juice:Carrot Juice:Grape Juice:Cancel")) {
			case 1:
				set .@fruit,512;
				set .@juice,531;
				break;
			case 2:
				set .@fruit,513;
				set .@juice,532;
				break;
			case 3:
				set .@fruit,515;
				set .@juice,534;
				break;
			case 4:
				set .@fruit,514;
				set .@juice,533;
				break;
			case 5:
				mes "[Mercador Marx Hansen]";
				mes "Muito bem, entăo...";
				mes "Até a próxima.";
				close;
			}
			if (countitem(.@fruit) == 0 || countitem(713) == 0 || Zeny < 3) {
				mes "[Mercador Marx Hansen]";
				mes "Oh, năo...";
				mes "You don't have all the necessary materials. To make "+getitemname(.@juice)+", I need 1 "+getitemname(.@fruit)+" and 1 Empty Bottle. I will also need a 3 zeny fee.";
				next;
				mes "[Mercador Marx Hansen]";
				mes "Quando vocę tiver preparado tudo, eu baterei a fruta para lhe dar um suco delicioso.";
				close;
			}

			mes "[Mercador Marx Hansen]";
			mes "Quantos vocę gostaria?";
			next;
			switch(select("As many as I can.:I want a certain amount.:Cancel.")) {
			case 1:
				set .@make,countitem(.@fruit);
				if (countitem(713) < .@make) set .@make, countitem(713);
				if (Zeny/3 < .@make) set .@make, Zeny/3;
				break;
			case 2:
				mes "[Mercador Marx Hansen]";
				mes "Choose a number less than 100. If you don't want to, put '0'. You can make up to " + countitem(.@fruit) + " bottles of juice.";
				next;
				while(1) {
					input .@input;
					if (.@input == 0) {
						mes "[Mercador Marx Hansen]";
						mes "Muito bem, entăo...";
						mes "Volte sempre.";
						close;
					}
					else if (.@input > 100) {
						mes "[Mercador Marx Hansen]";
						mes "Mais de 100 garrafas é impossível. Escolha uma quantidade diferente.";
						next;
					}
					else {
						break;
					}
				}
				set .@make,.@input;
				break;
			case 3:
				mes "[Mercador Marx Hansen]";
				mes "Muito bem, entăo...";
				mes "Volte sempre.";
				close;
			}

			set .@total_zeny,3 * .@make;

			if (countitem(.@fruit) < .@make || countitem(713) < .@make || Zeny < .@total_zeny) {
				mes "[Mercador Marx Hansen]";
				mes "Oh, năo...";
				mes "Vocę năo tem todos os materiais necessários. Năo posso ajudar em uma situaçăo como esta. Suponho que vocę deva coletar o que precisa.";
				close;
			}
			delitem .@fruit,.@make;
			delitem 713,.@make;
			set Zeny, Zeny-.@total_zeny;
			getitem .@juice,.@make;

			mes "[Mercador Marx Hansen]";
			mes "Aqui está! Suco fresco e delicioso como prometido. Deve ser bem refrescante e saboroso.";
			next;
			mes "[Mercador Marx Hansen]";
			mes "Muito bem, entăo...";
			mes "Volte sempre.";
			close;
		case 2:
			mes "[Mercador Marx Hansen]";
			mes "Antes que os humanos pudessem desenvolver sociedades tăo vastas, eles coletavam frutas das árvores para sobreviver. As frutas podem ter sido a bęnçăo da natureza que nos permitiu existir no mundo.";
			next;
			mes "[Mercador Marx Hansen]";
			mes "Como a vida se tornou tăo próspera, a geraçăo mais jovem parece năo comer mais frutas. Entăo, comecei a pensar em maneiras de tornar as frutas mais fáceis de comer.";
			next;
			mes "[Mercador Marx Hansen]";
			mes "Percebi que, quando se faz suco de frutas, é mais conveniente para consumir e tem um sabor muito melhor.";
			next;
			mes "# Informaçőes sobre Suco de Frutas #";
			mes "^CC4E5C- Suco de Maçă -^000000";
			mes "Maçă x 1 un, Garrafa Vazia x 1 un, 3 zenys.";
			mes "^E3CF57- Suco de Banana -^000000";
			mes "Banana x 1 un, Garrafa Vazia x 1 un, 3 zenys.";
			mes "^ED9121- Suco de Cenoura -^000000";
			mes "Cenoura x 1 un, Garrafa Vazia x 1 un, 3 zenys.";
			mes "^CC00FF- Suco de Uva -^000000";
			mes "Uva x 1 un, Garrafa Vazia x 1 un, 3 zenys.";
			close;
		case 3:
			mes "[Mercador Marx Hansen]";
			mes "Ei!";
			mes "Se vocę visitar";
			mes "alguém, fale com eles!";
			close;
		}
	}
	else {
		mes "[Mercador Marx Hansen]";
		mes "Bem-vindo.";
		mes "Como posso ajudá-lo?";
		next;
		if (select("Talk:Cancel") == 1) {
			mes "[Mercador Marx Hansen]";
			mes "Antes que os humanos pudessem desenvolver sociedades tăo vastas, eles coletavam frutas das árvores para sobreviver. As frutas podem ter sido a bęnçăo da natureza que nos permitiu existir no mundo.";
			next;
			mes "[Mercador Marx Hansen]";
			mes "Como a vida se tornou tăo próspera, a geraçăo mais jovem parece năo comer mais frutas. Entăo, comecei a pensar em maneiras de tornar as frutas mais fáceis de comer.";
			next;
			mes "[Mercador Marx Hansen]";
			mes "Percebi que, quando se faz suco de frutas, é mais conveniente para consumir e tem um sabor muito melhor.";
			close;
		}
		mes "[Mercador Marx Hansen]";
		mes "Ei!";
		mes "Se vocę visitar";
		mes "alguém, fale com eles!";
		close;
	}
