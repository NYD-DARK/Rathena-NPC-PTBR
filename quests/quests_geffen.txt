// Number Match Game :: gef_yagu
//============================================================
geffen,122,77,4	script	Nia#yagu	700,{
	mes "[Nia]";
	mes "Olá, sou Nia, a quinta";
	mes "discípula de Mawong, a";
	mes "mente brilhante dos jogos. Agora,";
	mes "como vocę gostaria de jogar um";
	mes "jogo de Combinaçăo de Números comigo?";
	next;
	switch(select("Jogo de Combinaçăo de Números?:Praticar Combinaçăo de Números:Jogar Combinaçăo de Números:Recordista:Cancelar")) {
	case 1:
		mes "[Nia]";
		mes "Ah, bem, o objetivo do";
		mes "jogo de Combinaçăo de Números é";
		mes "adivinhar o número em que estou pensando.";
		mes "É claro, existem algumas";
		mes "regras que limitam os números";
		mes "que posso escolher.";
		next;
		mes "[Nia]";
		mes "Todos os números que eu criar";
		mes "devem ter 3 dígitos, năo posso";
		mes "usar 0 em nenhum dos dígitos,";
		mes "e năo posso usar o mesmo";
		mes "número em nenhum dos dígitos";
		mes "mais de uma vez.";
		next;
		mes "[Nia]";
		mes "Entăo, por exemplo, năo posso";
		mes "usar os números 103, 112,";
		mes "252 ou 701. Esses números";
		mes "tęm um 0, ou usam";
		mes "o mesmo número nos dígitos";
		mes "mais de uma vez. Entendido?";
		next;
		mes "[Nia]";
		mes "Agora, vocę só tem 5 tentativas,";
		mes "e após cada palpite, direi";
		mes "quantos dígitos vocę acertou";
		mes "corretamente, seguidos pelo número de dígitos adivinhados corretamente na";
		mes "posiçăo correta do dígito.";
		next;
		mes "[Nia]";
		mes "Por exemplo, digamos que o";
		mes "número que escolhi seja 168.";
		mes "Se seu primeiro palpite for 678, entăo vocę acertou dois dos dígitos,";
		mes "6 e 8, corretamente. No entanto,";
		mes "apenas o 8 está na posiçăo correta.";
		next;
		mes "[Nia]";
		mes "Sabe, seria uma boa ideia";
		mes "se vocę tentasse o Modo Prática";
		mes "primeiro. Entăo, quando estiver mais";
		mes "confortável, jogue o";
		mes "jogo de Combinaçăo de Números real. Năo se esqueça que a velocidade conta para sua pontuaçăo!";
		next;
		mes "[Nia]";
		mes "Ah, se por acaso vocę";
		mes "fizer um novo recorde no";
		mes "jogo de Combinaçăo de Números, vocę";
		mes "terá seu nome registrado,";
		mes "entăo năo perca esta chance de";
		mes "fama e talvez até fortuna~";
		close;
	case 2:
		mes "[Nia]";
		mes "Ótimo, vamos fazer um jogo de";
		mes "Combinaçăo de Números de prática. Por favor,";
		mes "tente adivinhar meu número, ok?";
		mes "Se quiser cancelar, por favor, insira um número maior que 1.000.";
		set .@practice,1;
		break;
	case 3:
		mes "[Nia]";
		mes "Ótimo, vamos jogar Combinaçăo de Números!";
		mes "Vocę terá 5 chances de adivinhar o número em que estou pensando.";
		mes "Se quiser cancelar, por favor, insira um número maior que 1.000.";
		set .@practice,0;
		break;
	case 4:
		set .@name_record$, $050908_minus1_yagu$;
		if (.@name_record$ == "") //custom
			set .@name_record$,"(null)";
		set .@score_record, $050908_minus1_yagu;
		set .@score_min, .@score_record / 60;
		set .@score_sec, .@score_record % 60;
		mes "[Nia]";
		mes "^ff0000" + .@name_record$ + "^000000 is the";
		mes "detentor do recorde do jogo de";
		mes "Combinaçăo de Números com um tempo de";
		mes "^ff0000" + .@score_min + " minutes, " + .@score_sec + " seconds^000000.";
		close;
	case 5:
		mes "[Nia]";
		mes "Cuide-se,";
		mes "aventureiro. Se vocę algum dia";
		mes "sentir vontade de jogar um";
		mes "jogo de Combinaçăo de Números, venha falar";
		mes "comigo, certo? Até mais~";
		close;
	}
	next;

	// Generate 3 digits of a number.
	while(1) {
		set .@yagu100, rand(1,9);
		set .@yagu10, rand(1,9);
		set .@yagu1, rand(1,9);
		if ((.@yagu100 != .@yagu10) && (.@yagu100 != .@yagu1) && (.@yagu10 != .@yagu1)) {
			// Store the actual number (unofficial).
			set .@yagu, (100*.@yagu100) + (10*.@yagu10) + .@yagu1;
			break;
		}
	}

	// The official time calculation is based on ServerTime,
	// an int HHMMSS equivalent to atoi(gettimestr("%H%M%S",7)).
	// UNIX time is really a better way to do this.
	set .@start_time, gettimetick(2);

	while(1) {

		// Guess a number that meets all conditions.
		while(1) {
			if (input(.@input,100,999) != 0) {
				mes "[Nia]";
				mes "Vocę cancelou o";
				mes "jogo de Combinaçăo de Números.";
				close;
			}
			set .@input100, .@input / 100;
			set .@input10, (.@input % 100) / 10;
			set .@input1, .@input % 10;
			if ((.@input100 > 0) && (.@input10 > 0) && (.@input1 > 0)) {
				if ((.@input100 != .@input10) && (.@input100 != .@input1) && (.@input10 != .@input1))
					break;
			}
			mes "[Nia]";
			mes "Vocę inseriu um dos";
			mes "dígitos como 0, ou inseriu";
			mes "o mesmo valor numérico nos";
			mes "dígitos mais de uma vez.";
			mes "Por favor, insira outro palpite.";
			next;
		}

		set .@retry, .@retry+1;
		mes "[Nia]";
		mes "^ff0000"+strcharinfo(0)+"^000000,";
		mes "seu palpite é ^0000ff"+.@input+"^000000.";
		mes "Me dę um momento para";
		mes "apurar seus resultados.";
		next;

		// Correct digits in correct locations.
		set .@strike,0;
		if (.@yagu100 == .@input100)
			set .@strike, .@strike+1;
		if (.@yagu10 == .@input10)
			set .@strike, .@strike+1;
		if (.@yagu1 == .@input1)
			set .@strike, .@strike+1;

		// Correct digits in incorrect locations.
		set .@ball,0;
		if ((.@yagu100 == .@input10) || (.@yagu100 == .@input1))
			set .@ball, .@ball+1;
		if ((.@yagu10 == .@input100) || (.@yagu10 == .@input1))
			set .@ball, .@ball+1;
		if ((.@yagu1 == .@input100) || (.@yagu1 == .@input10))
			set .@ball, .@ball+1;

		if (.@strike == 3) {
			if (.@practice) {
				mes "[Nia]";
				mes "Sim, o número que";
				mes "o que eu palpitei foi ^ff0000"+.@yagu+"^000000!";
				mes "Parabéns, vocę";
				mes "acaba de vencer este jogo";
				mes "de prática! Agora vocę deve estar";
				mes "pronto para o desafio real~";
				close;
			} else {

				// Calculate final time and check against the previous record.
				set .@end_time, gettimetick(2);
				if (.@end_time < .@start_time)
					set .@end_time, .@end_time + 3600;
				set .@total_time, .@end_time - .@start_time;
				set .@total_min, .@total_time / 60;
				set .@total_sec, .@total_time % 60;
				set .@name_record$, $050908_minus1_yagu$;
				if (.@name_record$ == "") //custom
					set .@name_record$,"(null)";
				set .@score_record, $050908_minus1_yagu;
				set .@score_min, .@score_record / 60;
				set .@score_sec, .@score_record % 60;

				if (.@score_record >= .@total_time || !.@score_record) { //custom (second condition)
					set $050908_minus1_yagu$, strcharinfo(0);
					set $050908_minus1_yagu, .@total_time;
					mes "[Nia]";
					mes "Ótimo! Vocę adivinhou";
					mes "meu número, que era";
					mes "^ff0000" + .@yagu + "^000000. You finished in";
					mes "^ff0000" + .@total_min + " minutes, " + .@total_sec + " seconds^000000,";
					mes "quebrando o recorde anterior";
					mes "de ^ff0000" + .@score_min + " minutes, " + .@score_sec + " seconds^000000.";
					next;
					mes "[Nia]";
					mes "^ff0000" + .@name_record$ + "^000000";
					mes "estabeleceu esse antigo recorde,";
					mes "mas agora vocę é a";
					mes "nova pessoa a ser batida no";
					mes "Jogo de Combinaçăo. Parabéns";
					mes "por estabelecer um novo recorde!";
					close;
				}
				mes "[Nia]";
				mes "Bom trabalho, vocę adivinhou";
				mes "meu número! A resposta, claro,";
				mes "percurso, foi ^ff0000" + .@yagu + "^000000. You managed";
				mes "para adivinhar este número corretamente";
				mes "em ^ff0000" + .@total_min + " minutes, " + .@total_sec + " seconds^000000.";
				next;
				mes "[Nia]";
				mes "^ff0000" + .@name_record$ + "^000000 is";
				mes "o atual detentor do recorde";
				mes "do Combinaçăo de Números com um tempo";
				mes "de ^ff0000" + .@score_min + " minutes, " + .@score_sec + " seconds^000000.";
				mes "Boa sorte, e espero que vocę";
				mes "possa quebrar esse recorde algum dia~";
				close;
			}
		} else {
			if (.@practice) {
				mes "[Nia]";
				mes "Seu palpite tem ^ff0000" + .@ball + "^000000 of the";
				mes "números corretos. ^ff0000" + .@strike + "^000000 of the";
				mes "dígitos em seu palpite tęm";
				mes "o número correto na";
				mes "posiçăo correta do dígito.";
				mes "Muito bem, tente novamente~";
				next;
			} else {
				mes "[Nia]";
				mes "Seu último palpite teve ^ff0000" + .@ball + "^000000";
				mes "dos números corretos,";
				mes "e ^ff0000" + .@strike + "^000000 digits in your guess";
				mes "tinham o número correto na";
				mes "posiçăo correta do dígito.";
				switch(.@retry) {
					case 1: mes "Vocę tem quatro tentativas restantes."; break;
					case 2: mes "Vocę tem tręs tentativas restantes."; break;
					case 3: mes "Vocę tem duas tentativas restantes."; break;
					case 4: mes "Vocę tem apenas uma tentativa restante..."; break;
					case 5: mes "A resposta era ^ff0000" + .@yagu + "^000000."; close;
				}
				next;
			}
		}
	}
}
