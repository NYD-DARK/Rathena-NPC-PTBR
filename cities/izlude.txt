-	script	::Sailor_izlude	-1,{
	mes "[Marinheiro]";
	mes "Olá a todos!";
	mes "Atençăo, atençăo!";
	mes "Venham aproveitar o vento";
	mes "em um navio fascinante!";
	mes "Depressa, depressa!";
	next;
	switch(select("Byalan Island -> 150 Zeny.:Alberta Marina  -> 500 Zeny.:Cancel.")) {
	case 1:
		if (Zeny < 150) {
			mes "[Marinheiro]";
			mes "150 Zenys!";
			mes "Apenas 150 Zenys para viajar!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[Marinheiro]";
			mes "500 Zenys!";
			mes "Apenas 500 Zenys para viajar!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
