bat_room,124,178,5	script	Tierra Gorge Officer#01a	418,{
	if (getbattleflag("feature.bgqueue"))
		end;
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto !! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais com vocę. -";
		mes "- Por favor, tente novamente -";
		mes "- após reduzir um pouco do seu peso. -";
		close;
	}
	mes "[Oficial do Exército Guillaume]";
	mes "Vamos mostrar o poder do Exército Guillaume para aqueles Croixs fedorentos!";
	next;
	switch(select("I want to join your army!:End Conversation")) {
	case 1:
		if ((BaseJob == Job_Novice) || (BaseJob == Job_Super_Novice)) {
			mes "[Oficial do Exército Guillaume]";
			mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no cruel campo de batalha.";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Oficial do Exército Guillaume]";
				mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no cruel campo de batalha.";
			}
			else {
				set .@chk_urtime,checkquest(2069,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Oficial do Exército Guillaume]";
					mes "Vocę parece ter acabado de retornar do campo de batalha.";
					mes "É muito cedo para vocę voltar. Vá descansar e deixe os inimigos conosco!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2069;
					set .@mapcount,getmapusers("bat_a01");
					if (.@mapcount > 0) {
						mes "[Oficial do Exército Guillaume]";
						mes "Recebi um relatório informando que uma tropa de elite já foi enviada ao campo de batalha para lutar contra os Croixs.";
						mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
					}
					else {
						if ($@TierraBG1) {
							mes "[Oficial do Exército Guillaume]";
							mes "Uma tropa de elite já está de prontidăo para ser enviada ao campo de batalha.";
							mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
						}
						else {
							mes "[Oficial do Exército Guillaume]";
							mes "Vocę definitivamente parece estar pronto para a batalha!";
							mes "Vá mostrar aos Croixs o que o medo realmente significa!";
							mes "Hoje, nosso grito de vitória ecoará por todo o campo de batalha!";
							close2;
							warp "bat_room",57,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Oficial do Exército Guillaume]";
		mes "Hoje, seremos vitoriosos!";
		break;
	}
	close;
}

	next;

	mes "O exército que capturar a bandeira neutra no centro do campo de batalha será recompensado com pontos extras de regeneraçăo, o que significa que seus soldados ressuscitarăo mais rápido que o outro lado, garantindo uma vantagem.";
	next;
	mes "[Cavaleiro Guillaume]";
	mes "Esses dois depósitos de mantimentos estăo conectados um ao outro por caminhos curtos e estreitos, mas os portőes estăo bloqueados por barricadas.";
	mes "Pense e mova-se o mais rápido que puder. A vitória do seu exército depende da sua contribuiçăo.";
	next;
	mes "[Cavaleiro Guillaume]";
	if (getbattleflag("feature.bgqueue"))
		mes "Se estiver pronto, comece clicando no ^0000FFbotăo de batalha na interface^000000.";
	else {
		mes "Por favor, lembre-se: ao entrar em uma batalha, vocę receberá uma medalha que indica o tempo definido durante o qual vocę năo poderá participar do mesmo tipo de batalha.";
		mes "You may check the duration by pressing the Alt+U keys.";
	}
	next;
	mes "[Cavaleiro Guillaume]";
	if (getbattleflag("feature.bgqueue"))
		mes "Por favor, tenha em mente que, assim que cancelar, ^0000FFvocę năo poderá se inscrever novamente por 1 minuto^000000!";
	else
		mes "Está pronto para a batalha? Entăo inscreva-se com o recrutador ao meu lado!";
	close;
}
