// Flavius Officer - Guillaume
//============================================================
bat_room,133,178,5	script	Flavius Officer#01a	418,{
	if (getbattleflag("feature.bgqueue"))
		end;
	if (checkweight(1201,1) == 0) {
		mes "- Espere um momento!! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais com vocę. -";
		mes "- Por favor tente novamente -";
		mes "- depois de aliviar o peso. -";
		close;
	}
	mes "[Guillaume Army Officer]";
	mes "Vamos mostrar o poder do Exército Guillaume para esses Croixs fedorentos!";
	next;
	switch(select("I want to join your army!:End Conversation")) {
	case 1:
		if ((BaseJob == Job_Novice) || (BaseJob == Job_Super_Novice)) {
			mes "[Guillaume Army Officer]";
			mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso mandar um novato como vocę morrer no cruel campo de batalha.";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Guillaume Army Officer]";
				mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso mandar um novato como vocę morrer no cruel campo de batalha.";
			}
			else {
				set .@chk_urtime,checkquest(2070,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Guillaume Army Officer]";
					mes "Parece que vocę acabou de voltar do campo de batalha.";
					mes "É cedo demais para vocę voltar. Vá descansar e deixe os inimigos conosco!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2070;
					set .@mapcount2,getmapusers("bat_b01");
					if (.@mapcount2 > 0) {
						mes "[Guillaume Army Officer]";
						mes "Recebi um relatório informando que um corpo de elite já foi enviado ao campo de batalha para lutar contra os Croixs.";
						mes "Que tal esperar um pouco? Sugiro que afile suas armas e prepare seus suprimentos até lá.";
					}
					else {
						if ($@FlaviusBG1 > 0) {
							mes "[Guillaume Army Officer]";
							mes "Um corpo de elite já está de prontidăo para ser enviado ao campo de batalha.";
							mes "Que tal esperar um pouco? Sugiro que afile suas armas e prepare seus suprimentos até lá.";
						}
						else {
							mes "[Guillaume Army Officer]";
							mes "Vocę definitivamente parece estar pronto para a batalha!";
							mes "Vá mostrar aos Croixs o que o medo realmente significa!";
							mes "Hoje, nosso grito de vitória ecoará por todo o campo de batalha!";
							close2;
							warp "bat_room",85,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Guillaume Army Officer]";
		mes "Hoje, seremos vitoriosos!";
		break;
	}
	close;
}

		mes "Se estiver pronto, comece clicando no ^0000FFbotăo de batalha da interface^000000.";
	else {
		mes "Lembre-se: ao entrar em uma batalha, vocę receberá um token que indica o período pelo qual năo poderá participar do mesmo tipo de batalha.";
		mes "You may check the duration by pressing the Alt+U keys.";
	}
	next;
	mes "[Guillaume Knight]";
	if (getbattleflag("feature.bgqueue"))
		mes "Tenha em mente que, ao cancelar, ^0000FFvocę năo poderá se reinscrever por 1 minuto^000000!";
	else
		mes "Está pronto para a batalha? Entăo se inscreva com o recrutador ao meu lado!";
	close;
}
