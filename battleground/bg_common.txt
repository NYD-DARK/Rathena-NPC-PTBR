			set .@menu1$,"Gunslinger:Swordman/Taekwon Master:Thief:Acolyte:Magician:Archer:Merchant";
			break;
		case 4:
			mes "[Erundek]";
			mes "Vocę escolheu ^3131FFConsumível^000000.";
			mes "Os seguintes itens consumíveis estăo disponíveis para troca com os emblemas do campo de batalha:";
			next;
			setarray .@Items[0],12269,10,12270,10,12271,5,12272,10,12273,10;
			break;
		}
		break;
	case 2:
		mes "[Erundek]";
		mes "Temos muitos itens, entăo por favor dę uma olhada e compre com cautela.";
		close2;
		readbook 11010,1;
		end;
	}
	set .@menu$,"";
	if (.@menu1$ != "") set .@menu$, .@menu1$;
	else for(set .@i,0; .@i<getarraysize(.@Items); set .@i,.@i+2)
		set .@menu$, .@menu$+getitemname(.@Items[.@i])+":";
	set .@i, (select(.@menu$)-1)*2;
	mes "[Erundek]";
	mes "Vocę escolheu ^3131FF"+getitemname(.@Items[.@i])+"^000000.";
	switch(.@Items[.@i]) {
		case 2720: mes "Este item é apenas para a Classe Espadachim e Mestre Taekwon."; break;
		case 2721: mes "Este item é apenas para a Classe Gatuno."; break;
		case 2722: mes "Este item é apenas para a Classe Noviço."; break;
		case 2723: mes "Este item é apenas para a Classe Mago."; break;
		case 2724: mes "Este item é apenas para a Classe Arqueiro."; break;
		case 2725: mes "Este item é apenas para a Classe Mercador."; break;
		case 2733: mes "Este item é apenas para Justiceiro."; break;
		default: break;
	}
	mes "Vocę pode trocar por este item com ^FF0000"+.@Items[.@i+1]+" "+getitemname(7828)+" or "+.@Items[.@i+1]+" "+getitemname(7829)+"^000000.";
	mes "Gostaria de realizar a troca?";
	next;
	switch(select("Năo trocar:Trocar")) {
	case 1:
		mes "[Erundek]";
		mes "Vocę precisa de mais tempo para verificar os itens?";
		break;
	case 2:
		mes "[Erundek]";
		mes "Qual Emblema vocę quer trocar?";
		mes "Vocę precisa de ^3131FF"+.@Items[.@i+1]+" Badges^000000 to exchange.";
		next;
		if (.@item[0] < 12269 || .@item[0] > 12273 ) {
			mes "[Erundek]";
			mes "Lembre-se, os Itens de Recompensa do Campo de Batalha săo ^FF0000Presos ao Personagem^000000. Tem certeza de que quer este item?";
			next;
		}
		set .@j, select("Emblema de Bravura:Emblema de Valor:Cancelar");
		mes "[Erundek]";
		if (.@j == 3) {
			mes "Vocę cancelou a troca.";
			break;
		}
		set .@cost, ((.@j==1)?7828:7829);
		if (countitem(.@cost) >= .@Items[.@i+1]) {
			mes "Obrigado pela troca.";
			delitem .@cost, .@Items[.@i+1];
			getitem .@Items[.@i],1;
		}
		else mes "Vocę năo tem "+getitemname(.@cost)+"s.";
		break;
	}
	close;
}

// Badge Reward Function
//============================================================
function	script	F_BG_Badge	{
	.@team$ = getarg(1);
	.@bg_type$ = getarg(2);

	if (.@bg_type$ == "Tierra") {
		set .@badge,7828; //BF_Badge1
		set .@amount_win,3;
		set .@amount_lose,1;
	} else if (.@bg_type$ == "Flavius") {
		set .@badge,7829; //BF_Badge2
		set .@amount_win,9;
		set .@amount_lose,3;
	} else
		return;

	if (getarg(0) == 1) {
		set .@amount, .@amount_win;
		if (.@team$ == "Guillaume") {
			mes "[Axl Rose]";
			mes "Abençoado seja Guillaume!";
		} else if (.@team$ == "Croix") {
			mes "[Swandery]";
			mes "Abençoado seja Croix!";
		}
		mes "Vamos desfrutar da nossa gloriosa vitória!";
		mes strcharinfo(0)+", it's a sign reflecting victory.";
	} else {
		set .@amount, .@amount_lose;
		if (.@team$ == "Guillaume") {
			mes "[Axl Rose]";
			mes "Vocę perdeu, mas se dedicou a esta batalha.";
			mes "Esta é uma recompensa por sua grande dedicaçăo, dada por Guillaume Marollo!";
			mes "Apenas encare esta derrota como uma liçăo, e na próxima vez vocę com certeza vencerá.";
		} else if (.@team$ == "Croix") {
			mes "[Swandery]";
			mes "Oh, "+strcharinfo(0)+" Don't be sad.";
			mes "Mesmo que năo tenhamos vencido, demos o nosso melhor.";
			mes "Este é um presente Real de Croix, e por favor năo se esqueça desta batalha. Venceremos a próxima.";
		}
	}
	close2;

	// If the VIP system is enabled, VIP players receive an extra 2 badges.
	if (VIP_SCRIPT && vip_status(VIP_STATUS_ACTIVE))
		set .@amount, .@amount+2;

	set .@medal_gap, 500 - countitem(.@badge);
	if (.@medal_gap >= .@amount)
		getitem .@badge,.@amount;
	else
		getitem .@badge,.@medal_gap;
	return;
}
