	mes "[Logística]";

	if (!.@item_type)
		mes "Vocę selecionou ^ff0000"+ getitemname(.@item_id) +"^000000.";
	else if (.@item_type == 1)
		mes "Existe apenas um "+ callfunc("F_GetWeaponType",.@item_id) +": "+ getitemname(.@item_id) +".";
	else if (.@item_type == 2) {
		mes "Vocę selecionou '"+ callfunc("F_GetArmorType",.@item_id) +"'.";
		mes "A armadura para quem tem Nível acima de 80 é: ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}
	else if (.@item_type == 3) {
		mes "Vocę selecionou 'Produçăo em Massa "+ callfunc("F_GetArmorType",.@item_id) +"'.";
		mes "Existe apenas um(a) Produçăo em Massa "+ callfunc("F_GetArmorType",.@item_id) +": ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}
	else if (.@item_type == 4) {
		mes "Vocę selecionou 'Popularizado(a) "+ callfunc("F_GetArmorType",.@item_id) +"'.";
		mes "Existe apenas um(a) Popularizado(a) "+ callfunc("F_GetArmorType",.@item_id) +": ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}

	if (!.@req_setting)
		mes "Requer ^0000ff2.000^000000 Pontos KVM, e "+ .@name$ +", your points are: ^580080"+ kvm_point +"^000000.";

	next;
	switch(select("Năo, năo vou comprar.:Sim, vou comprar.")) {
	case 1:
		mes "[Logística]";
		mes "Vocę selecionou 'Năo vou comprar'.";
		mes "Ao comprar um item, por favor, tome cuidado, pois năo há reembolsos.";
