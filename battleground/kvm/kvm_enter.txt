// KvM Officer - Guillaume
//============================================================
bat_room,164,178,5	script	KVM Mercenary Officer#1	418,{
	if (getbattleflag("feature.bgqueue"))
		end;
	/*
	set .@kvm,checkquest(6026,PLAYTIME);
	if ((.@kvm == 0) || (.@kvm == 1)) {
		mes "[Oficial Mercenário Croix]";
		mes "Eu sei que vocę já assinou com os Croix. Volte e junte-se ŕ equipe deles. Năo aceitaremos traidores lutando por nós.";
		close;
	}
	else if (.@kvm == 2) erasequest 6026;
	*/
	set .@permit,checkquest(6025,PLAYTIME);
	if ((.@permit == 0) || (.@permit == 1)) {
		mes "[Oficial Mercenário Guillaume]";
		mes "Parece que vocę participou recentemente de uma batalha da KVM. Vocę deve esperar 5 minutos antes de se inscrever novamente.";
		close;
	}
	else if (.@permit == 2) erasequest 6025;
	mes "[Oficial Mercenário Guillaume]";
	mes "Mostre a eles o verdadeiro poder de Guillaume!";
	next;
	switch(select("I will fight with you.:End Conversation.")) {
	case 1:
		mes "[Oficial Mercenário Guillaume]";
		mes "Mostre a eles o quăo fortes nós somos.";
		mes "Hoje, todos ouvirăo o grito de triunfo de Guillaume!";
		close2;
		//setquest 6025;
		if (BaseLevel > 79)  warp "bat_room",169,223;
		else if (BaseLevel < 60) warp "bat_room",197,223;
		else warp "bat_room",225,223;
		end;
	case 2:
		mes "[Oficial Mercenário Guillaume]";
		mes "Nós vamos vencer!";
		close;
	}
}

// Guillaume Knight - KvM
//============================================================
bat_room,167,178,5	script	Guillaume Knight#kvm	417,{
	mes "[Cavaleiro Guillaume]";
	mes "Olá.";
	mes "O que vocę gostaria de saber?";
	next;
	switch(select("Apply for KVM.:What is KVM?:How do I participate in KVM?:I want to know my Kreiger Points.")) {
	case 1:
		mes "[Cavaleiro Guillaume]";
		if (getbattleflag("feature.bgqueue"))
			mes "Se estiver pronto para a batalha, clique no ^0000FFbotăo de batalha na interface^000000 para se inscrever na batalha da KVM!";
		else {
			mes "As inscriçőes ainda năo estăo disponíveis.";
			mes "Para se inscrever, vocę precisa ir até um Oficial Mercenário da KVM.";
		}
		close;
	case 2:
		mes "[Cavaleiro Guillaume]";
		mes "KVM é a abreviaçăo de Krieger Von Midgard.";
		mes "Aventureiro, vocę está ciente de que o caminho para o novo world foi aberto?";
		next;
		mes "[Cavaleiro Guillaume]";
		mes "Nós, da Administraçăo Guillaume, temos vários planos para avançar rumo ao novo mundo.";
		mes "E a KVM é um deles.";
		next;
		mes "[Cavaleiro Guillaume]";
		mes "Isso significa que nós, uma das muitas naçőes do continente de Midgard,";
		mes "decidimos recrutar muitos aventureiros para o avanço imediato ao novo mundo.";
		next;
		mes "[Cavaleiro Guillaume]";
		mes "Portanto, para selecionar os melhores aventureiros, estamos realizando a KVM.";
		mes "Esperamos a participaçăo de muitos aventureiros.";
		next;
		mes "[Cavaleiro Guillaume]";
		mes "E nós daremos recompensas pela participaçăo deles!";
		mes "Quanto ŕs recompensas, por favor, entre em contato com um Oficial de Logística da KVM.";
		close;
	case 3:
		mes "[Cavaleiro Guillaume]";
		if (getbattleflag("feature.bgqueue")) {
			mes "Basicamente, a KVM é uma batalha de 3 contra 3.";
			mes "Inscreva-se clicando no ^0000FFbotăo de batalha na interface^000000, em grupo ou individualmente.";
		} else {
			mes "Basicamente, a KVM é uma batalha de 5 contra 5.";
			mes "Primeiro, vocę se inscreve com um recepcionista da KVM, em grupo ou individualmente.";
		}
		next;
		mes "[Cavaleiro Guillaume]";
		if (getbattleflag("feature.bgqueue"))
			mes "As inscriçőes em grupo săo para quando vocę pretende entrar na KVM com os membros do seu grupo ou clă,";
		else
			mes "As inscriçőes em grupo săo para quando vocę pretende entrar na KVM com os membros do seu grupo,";
		mes "e a inscriçăo individual é para quando vocę pretende entrar na KVM sozinho.";
		next;
		mes "[Cavaleiro Guillaume]";
		if (getbattleflag("feature.bgqueue"))
			mes "Por favor, note que se vocę cancelar, ^0000FFnăo poderá se inscrever novamente por 1 minuto^000000.";
		else {
			mes "Por favor, inscreva-se com um oficial da KVM, e ele entrará em contato com vocę mais tarde quando vocę estiver em Prontera.";
			next;
			mes "[Cavaleiro Guillaume]";
			mes "Entăo vocę entra e segue as instruçőes no campo de batalha.";
			next;
			mes "[Cavaleiro Guillaume]";
			mes "No entanto, esteja ciente de que, a menos que vocę esteja no escritório da KVM, ele năo poderá entrar em contato com vocę.";
		}
		close;
	case 4:
		mes "[Cavaleiro Guillaume]";
		mes "Seus Pontos de Krieger săo:";
		mes ""+kvm_point+".";
		close;
	}
}
