alberta,23,238,4	script	Guide#alb::AlbGuide	105,{
	cutin "prt_soldier",2;
	mes "[Guia de Alberta]";
	mes "Bem-vindo a Alberta,";
	mes "a Cidade Portuária. Sinta-se ŕ vontade";
	mes "para me perguntar caso tenha";
	mes "dificuldade em encontrar algo na";
	mes "cidade, ou se precisar apenas de";
	mes "orientaçăo pela cidade.";
	set .@loop1,1;
	while (.@loop1) {
		next;
		switch(select("Guia da Cidade:Remover Marcas do Mini-Mapa:Aviso:Cancelar")) {
		case 1:
			mes "[Guia de Alberta]";
			mes "Por favor, selecione";
			mes "um local a partir";
			mes "do menu a seguir.";
			if (.@compass_check == 0) {
				mes "Vocę gostaria que eu";
				mes "marcasse locais";
				mes "no seu Minimapa?";
				next;
				if (select("Sim:Năo") == 1) set .@compass_check,1;
			}
			set .@loop2,1;
			while (.@loop2) {
				if (.@wait_button_chk == 0) set .@wait_button_chk,1;
				else next;
				
				switch(select("^FF0000Guilda dos Mercadores^000000:Loja de Armas:Loja de Ferramentas:Pousada:Forja:Cancelar")) {
				case 1:
					mes "[Guia de Alberta]";
					mes "A Guilda dos Mercadores";
					mes "cuida das Mudanças de Classe";
					mes "para a Classe Mercador,";
					mes "e está localizada no";
					mes "canto sudoeste";
					mes "de Alberta.";
					if (.@compass_check)
						viewpoint 1,33,41,2,0xFF0000;
					break;
				case 2:
					mes "[Guia de Alberta]";
					mes "A Loja de Armas";
					mes "pode ser encontrada no";
					mes "extremo sul de Alberta.";
					if (.@compass_check)
						viewpoint 1,117,37,3,0xFF00FF;
					break;
				case 3:
					mes "[Guia de Alberta]";
					mes "A Loja de Utilidades";
					mes "fica bem próxima";
					mes "ao centro de";
					mes "Alberta. Năo deve";
					mes "ser difícil de encontrar.";
					if (.@compass_check)
						viewpoint 1,98,154,4,0xFF00FF;
					break;
				case 4:
					mes "[Guia de Alberta]";
					mes "Há uma Estalagem";
					mes "no extremo";
					mes "norte de Alberta";
					mes "onde vocę pode descansar.";
					if (.@compass_check)
						viewpoint 1,65,233,5,0xFF00FF;
					break;
				case 5:
					mes "[Guia de Alberta]";
					mes "A Forja em Alberta";
					mes "fica no mesmo prédio";
					mes "que a Guilda dos Mercadores.";
					mes "Fica ao sudoeste.";
					if (.@compass_check)
						viewpoint 1,35,41,6,0xFF00FF;
					break;
				case 6:
					mes "[Guia de Alberta]";
					mes "Por favor, peça-me para ''Remover";
					mes "Marcas do Minimapa'' se vocę";
					mes "năo quiser mais que as";
					mes "marcas de localizaçăo sejam exibidas";
					mes "no seu Minimapa.";
					set .@loop2,0;
				}
			}
			break;
		case 2:
			viewpoint 2,237,41,2,0xFF0000;
			viewpoint 2,237,41,3,0xFF00FF;
			viewpoint 2,46,345,4,0xFF00FF;
			viewpoint 2,175,220,5,0xFF00FF;
			viewpoint 2,175,220,6,0xFF00FF;
			set .@compass_check,0;
			break;
		case 3:
			mes "[Guia de Alberta]";
			mes "Avanços em feitiçaria e";
			mes "tecnologia nos permitiram";
			mes "atualizar nosso sistema de";
			mes "informaçőes, permitindo marcar";
			mes "locais no seu Minimapa";
			mes "para uma navegaçăo mais fácil.";
			next;
			mes "[Guia de Alberta]";
			mes "O seu Minimapa está localizado";
			mes "no canto superior direito";
			mes "da tela. Se năo conseguir";
			mes "para vę-lo, pressione Ctrl + Tab";
			mes "teclas ou clicar no botăo ''Mapa'' na sua Janela de Informaçőes Básicas.";
			next;
			mes "[Guia de Alberta]";
			mes "No seu Minimapa,";
			mes "clique nos botőes ''+'' e ''-''";
			mes "símbolos para dar zoom in e";
			mes "zoom out no seu Minimapa. Esperamos";
			mes "que vocę aproveite suas viagens";
			mes "aqui na cidade de Alberta.";
			break;
		case 4:
			mes "[Guia de Alberta]";
			mes "Fique seguro quando";
			mes "viajar e năo hesite";
			mes "em me perguntar caso tenha";
			mes "alguma dúvida sobre Alberta.";
			close2;
			set .@loop1,0;
		}
	}
	cutin "prt_soldier",255;
	end;
}
