function Dyes {
	setarray .@DyeName$[0],"the default color","Orange","Violet","Red","Black","Green","Blue","White","Yellow";
	setarray .@DyeItems[0],0,980,981,975,983,979,978,982,976;
	set .@menu$,"- Default:";
	for(set .@i,0; .@i<getargcount(); set .@i,.@i+2)
		set .@menu$, .@menu$+"- "+.@DyeName$[getarg(.@i)]+":";
	set .@menu$, .@menu$+"- ^777777Cancel^000000";
	set .@s, select(.@menu$)-1;
	if (.@s == (getargcount()/2)+1) goto L_End;
	mes .n$;
	if (.@s == 0) {
		mes "Posso mudar seu corante para";
		mes "o padrăo gratuitamente.";
		mes "Vocę tem certeza?";
		next;
		switch(select("Sim:Năo")) {
		case 1:
			mes .n$;
			setlook 7,0;
			mes "Suas roupas foram tingidas de "+.@DyeName$[0]+".";
			close;
		case 2:
			goto L_End;
		}
	}
	if (!countitem(.@DyeItems[getarg((.@s-1)*2)])) {
		mes "Para eu tingir suas roupas, precisarei dos corantes apropriados. Por favor, verifique minha lista de preços para mais informaçőes.";
		close;
	}
	if (Zeny < 10000) {
		mes "Sinto muito, mas vocę năo tem dinheiro suficiente.";
		close;
	}
	delitem .@DyeItems[getarg((.@s-1)*2)],1;
	set Zeny, Zeny-10000;
	setlook 7, ((.@s)?(getarg((.@s-1)*2+1)):0);
	mes "Suas roupas foram tingidas de "+((.@s)?.@DyeName$[getarg((.@s-1)*2)]:.@DyeName$[0])+".";
	close;
}
