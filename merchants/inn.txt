lhz_in02,230,284,4	script	Hotel Employee#01	86,{
	mes "[Funcionário do Hotel]";
	mes "Bem-vindo ao";
	mes "Royal Dragon,";

	mes "melhores acomodaçőes";

	case 1:
		savepoint "lhz_in02",209,275,1,1;
		mes "[Funcionário do Hotel]";
		mes "Obrigado, seu";
		mes "Ponto de Retorno foi";
		mes "salvo aqui no";
		mes "Royal Dragon.";
		close;
	case 2:
		if(Zeny < 5000) {
			mes "[Funcionário do Hotel]";
			mes "Sinto muito, mas";
			mes "vocę precisa de 5.000 zenys";
			mes "para fazer o check-in.";
			close;
		}
		set Zeny,Zeny - 5000;
		mes "[Funcionário do Hotel]";
		mes "Obrigado~";
		mes "Espero que aproveite";
		mes "sua estadia no";
		mes "Royal Dragon.";
		close2;
		warp "lhz_in02",219,150;
		percentheal 100,100;
		npcskill "AL_BLESSING",10,99,99;
		end;
	case 3:
		mes "[Funcionário do Hotel]";
		mes "Obrigado e";
		mes "tenha um bom dia.";
		close;
	}
}

bra_in01,27,24,3	script	Hotel Keeper#bra1	478,{
	mes "[Recepcionista do Hotel]";
	mes "Bem-vindo ao belo Hotel de Brasilis.";
	next;
	switch(select("Save:Rest -5000 zeny:Cancel")) {
	case 1:
		mes "[Recepcionista do Hotel]";
		mes "Vocę deseja salvar seu Ponto de Retorno aqui no Hotel de Brasilis?";
		next;
		switch(select("No thank you.:Absolutely.")) {
		case 1:
			mes "[Recepcionista do Hotel]";
			mes "Ok entăo, aproveite sua estadia.";
			close;
		case 2:
			mes "[Recepcionista do Hotel]";
			mes "Seu Ponto de Retorno foi salvo aqui no hotel. Espero que vocę aproveite sua estadia aqui em Brasilis.";
			savepoint "bra_in01",144,69,1,1;
			close;
		}
		end;
	case 2:
		if (Zeny > 4999) {
			mes "[Recepcionista do Hotel]";
			mes "Vou lhe mostrar um ótimo quarto.";
			close2;
			if (Zeny < 5000)
				end;
			Zeny -= 5000;
			percentheal 100,100;
			warp "bra_in01",144,69;
			end;
		}
		mes "[Recepcionista do Hotel]";
		mes "Sinto muito, mas a taxa de serviço é de 5.000 zenys por noite.";
	case 3:
		close;
	}
}

//======================= Inn Function ==============================
function	script	F_InnMaid	{
	.@npc_name$ = getarg(0);

	mes .@npc_name$;
	mes "Bem-vindo ao";
	mes getarg(1) + ".";
	mes "Como posso ajudar?";
	next;
	switch(select("Save:Take a Rest -> 5000 zeny:Cancel")) {
	case 1:
		mes .@npc_name$;
		mes "Seu Ponto de Retorno";
		mes "foi salvo.";
		mes "Obrigado,";
		mes "por favor, volte sempre.";
		savepoint getarg(2),getarg(3),getarg(4),1,1;
		close;
	case 2:
		mes .@npc_name$;
		if(Zeny < 5000){
			mes "Sinto muito, mas a taxa de serviço é de 5.000 zenys. Por favor, certifique-se de ter dinheiro suficiente para o check-in na próxima vez, ok?";
			close;
		}
		mes "Obrigado.";
		mes "Espero que vocę";
		mes "aproveite seu descanso~";
		close2;
		set Zeny,Zeny - 5000;
		percentheal 100,100;
		return;
	case 3:
		close;
	}
}
