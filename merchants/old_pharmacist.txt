alberta_in,16,28,4	script	Pharmacist	61,{
	if (checkweight(1201,1) == 0) {
		mes "- Espere um momento! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais com vocę. -";
		mes "- Por favor, volte mais tarde -";
		mes "- depois de colocar alguns itens no armazém da Kafra. -";
		close;
	}
	mes "[Velho Farmacęutico]";
	mes "Ummmm...";
	mes "O que traz vocę aqui...?";
	next;
	switch(select("Preparar Poçăo:Conversar.:Informaçőes de Mistura:Cancelar")) {
	case 1:
		if (MaxWeight - Weight < 5000) {
			mes "[Velho Farmacęutico]";
			mes "Por que vocę está carregando tudo isso!";
			mes "Năo seja ganancioso, carregue apenas o que precisa!";
			next;
			mes "[Velho Farmacęutico]";
			mes "Vocę está muito pesado para receber poçőes minhas...";
			mes "Vá guardar alguns itens no seu armazém primeiro!";
			close;
		} else {
			mes "[Velho Farmacęutico]";
			mes "Vocę está com todos os itens prontos, certo? Qual vocę gostaria?";
			next;
			switch(select("Poçăo Vermelha.:Poçăo Laranja.:Poçăo Amarela.:Poçăo Branca.:Poçăo Azul.:Poçăo Verde.:Na verdade, năo quero nada.")) {
			case 1:
				callsub L_Making,507,3,501;
			case 2:
				mes "[Velho Farmacęutico]";
				if (countitem(507) < countitem(713)) {
					set .@max,countitem(507);
				}
				else if (countitem(508) < countitem(713)) {
					set .@max,countitem(508);
				}
				else {
					set .@max,countitem(713);
				}
				if ((countitem(507) < 1) || (countitem(508) < 1) || (countitem(713) == 0)) {
					mes "Seu moleque! O que vocę esperava?! Vir aqui sem nada. Tsk!";
					mes "Desapareça!";
					close;
				}
				if (Zeny < 3) {
					mes "Seu moleque! O que vocę esperava?! Vir aqui sem nada. Tsk!";
					mes "Desapareça!";
					close;
				}
				mes "Quantas?";
				next;
				switch(select("Preparar o máximo que puder.:Quero escolher uma quantidade.:Na verdade, năo quero nada.")) {
				case 1:
					if ((countitem(507) < .@max) || (countitem(508) < .@max) || (countitem(713) < .@max) || (Zeny < .@max*3)) {
						mes "[Velho Farmacęutico]";
						mes "Seu moleque! Vocę nem tem todos os materiais e quer que eu faça poçőes para vocę?!";
						close;
					}
					set Zeny,Zeny-(.@max*5);
					delitem 507,.@max; //Red_Herb
					delitem 508,.@max; //Yellow_Herb
					delitem 713,.@max; //Empty_Bottle
					getitem 502,.@max; //Orange_Potion
					break;
				case 2:
					mes "[Velho Farmacęutico]";
					mes "Entăo escolha um número abaixo de 100. Se năo quiser nenhuma, apenas digite '0'. Com os materiais que vocę tem, pode fazer cerca de "+.@max+" potions.";
					next;
					input .@amount,0,101;
					if (.@amount == 0) {
						mes "[Velho Farmacęutico]";
						mes "Decida-se, quer fazer o favor?!";
						close;
					}
					if (.@amount > 100) {
						mes "[Velho Farmacęutico]";
						mes "Vocę é surdo? Eu disse menos de 100!";
						close;
					}
					if ((countitem(507) < .@amount) || (countitem(508) < .@amount) || (countitem(713) < .@amount) || (Zeny < .@amount*3)) {
						mes "[Velho Farmacęutico]";
						mes "Seu moleque! Vocę nem tem todos os materiais e quer que eu faça poçőes para vocę?!";
						close;
					}
					next;
					set Zeny,Zeny-(.@amount*5);
					delitem 507,.@amount; //Red_Herb
					delitem 508,.@amount; //Yellow_Herb
					delitem 713,.@amount; //Empty_Bottle
					getitem 502,.@amount; //Orange_Potion
					break;
				case 3:
					mes "[Velho Farmacęutico]";
					mes "O quę?!";
					mes "Grrr...";
					mes "Bleh!";
					mes "Desapareça!";
					close;
				}
				mes "[Velho Farmacęutico]";
				mes "Aqui está. Está tudo pronto, pode levar. Mas lembre-se! Abusar de medicamentos năo é bom.";
				close;
			case 3:
				callsub L_Making,508,10,503;
			case 4:
				callsub L_Making,509,20,504;
			case 5:
				callsub L_Making,510,30,505;
			case 6:
				callsub L_Making,511,3,506;
			case 7:
				mes "[Velho Farmacęutico]";
				mes "O quę?!";
				mes "Grrr...";
				mes "Bleh!";
				mes "Desapareça!";
				close;
			}
		}
	case 2:
		mes "[Velho Farmacęutico]";
		mes "Com medicamentos, vocę pode aumentar a capacidade de regeneraçăo de uma pessoa. Mas eles só funcionam até certo ponto. *Suspiro* Estou começando a pensar nos dias em que eu era jovem. Devo estar ficando velho.";
		next;
		mes "[Velho Farmacęutico]";
		mes "De qualquer forma, uma poçăo é apenas uma poçăo. Nada mais e nada menos.";
		close;
	case 3:
		mes "[Velho Farmacęutico]";
		mes "Hrrrmm...";
		mes "Vocęs jovens podem ser bem irritantes. Mas, já que perguntou, eu explico.";
		next;
		mes "[Velho Farmacęutico]";
		mes "Ervas funcionam bem por si só, mas se vocę usar minhas técnicas e habilidades especiais para fazer poçőes com elas, o efeito é muito, muito maior.";
		next;
		mes "[Velho Farmacęutico]";
		mes "Se vocę pedir com vontade e educaçăo, eu farei para vocę. Mas năo de graça... Năo se preocupe, eu só cobro uma pequena taxa, entăo năo é tăo caro.";
		next;
		mes "[Velho Farmacęutico]";
		mes "Poçăo Vermelha - ^0098E52 Ervas Vermelhas, 1 Garrafa Vazia, taxa de 2 Zenys.^000000";
		mes "Poçăo Laranja - ^0098E51 Erva Vermelha, 1 Erva Amarela, 1 Garrafa Vazia, taxa de 5 Zenys.^000000";
		mes "Poçăo Amarela - ^0098E52 Ervas Amarelas, 1 Garrafa Vazia, 10 Zenys.^000000";
		next;
		mes "[Velho Farmacęutico]";
		mes "Poçăo Branca - ^0098E52 Ervas Brancas, 1 Garrafa Vazia, taxa de 20 Zenys.^000000";
		mes "Poçăo Azul - ^0098E52 Ervas Azuis, 1 Garrafa Vazia, taxa de 30 Zenys.^000000";
		mes "Poçăo Verde - ^0098E52 Ervas Verdes, 1 Garrafa Vazia, taxa de 3 Zenys.^000000";
		close;
	case 4:
		mes "[Velho Farmacęutico]";
		mes "Que pessoa chata. Se vocę tem algo a dizer, por que năo diz?!";
		close;
	}
	
L_Making:
	.@item_req = getarg(0);
	.@req_amount = getarg(1);

	mes "[Velho Farmacęutico]";
	if ((countitem(.@item_req)/2) < countitem(713)) {
		.@max = countitem(.@item_req) /2;
	} else {
		.@max = countitem(713);
	}
	if (countitem(.@item_req) < 2 || countitem(713) == 0) {
		mes "Seu moleque! O que vocę esperava?! Vir aqui sem nada. Tsk!";
		mes "Desapareça!";
		close;
	}
	if (Zeny < 3) {
		mes "Seu moleque! O que vocę esperava?! Vir aqui sem nada. Tsk!";
		mes "Desapareça!";
		close;
	}
	mes "Quantas?";
	next;
	switch(select("Preparar o máximo que puder.:Quero escolher uma quantidade.:Na verdade, năo quero nada.")) {
	case 1:
		if ((countitem(.@item_req) < .@max*2) || countitem(713) < .@max || (Zeny < .@max*.@req_amount)) {
			mes "[Velho Farmacęutico]";
			mes "Seu moleque! Vocę nem tem todos os materiais e quer que eu faça poçőes para vocę?!";
			close;
		}
		Zeny = Zeny - (.@max*.@req_amount);
		delitem .@item_req,.@max*2;
		delitem 713,.@max; //Empty_Bottle
		getitem getarg(2),.@max; 
		break;
	case 2:
		mes "[Velho Farmacęutico]";
		mes "Entăo escolha um número abaixo de 100. Se năo quiser nenhuma, apenas digite '0'. Com os materiais que vocę tem, pode fazer cerca de "+.@max+" potions.";
		next;
		input .@amount;
		if (.@amount == 0) {
			mes "[Velho Farmacęutico]";
			mes "Decida-se, quer fazer o favor?!";
			close;
		}
		if (.@amount > 100) {
			mes "[Velho Farmacęutico]";
			mes "Vocę é surdo? Eu disse menos de 100!";
			close;
		}
		if ((countitem(.@item_req) < .@amount*2) || countitem(713) < .@amount || (Zeny < .@amount*.@req_amount)) {
			mes "[Velho Farmacęutico]";
			mes "Seu moleque! Vocę nem tem todos os materiais e quer que eu faça poçőes para vocę?!";
			close;
		}
		Zeny = Zeny - (.@max*.@req_amount);
		delitem .@item_req,.@amount*2;
		delitem 713,.@amount; //Empty_Bottle
		getitem getarg(2),.@amount; 
		break;
	case 3:
		mes "[Velho Farmacęutico]";
		mes "O quę?!";
		mes "Grrr...";
		mes "Bleh!";
		mes "Desapareça!";
		close;
	}
	mes "[Velho Farmacęutico]";
	mes "Aqui está. Está tudo pronto, pode levar. Mas lembre-se! Abusar de medicamentos năo é bom.";
	close;
}
