prontera,165,60,2	script	Apprentice Craftsman	73,{
	disable_items;
	mes "[Artesão Aprendiz]";
	if (Zeny >= 400000) {
		mes "Tenho estudado formas de aprimorar uma armadura para maximizar sua capacidade.";
		next;
		mes "[Artesão Aprendiz]";
		mes "Encantar é uma habilidade incrível que infunde poderes de status misteriosos no slot oculto da armadura.";
		next;
		mes "[Artesão Aprendiz]";
		mes "No entanto, você deve ter em mente que se houver duas armaduras do mesmo tipo em sua posse, o Encantamento será aplicado na ordem em que estiverem no seu inventário.";
		next;
		mes "[Artesão Aprendiz]";
		mes "Nesse caso, o Encantamento pode ser aplicado a um item que você não pretendia encantar. Então, para garantir, traga apenas ^5555ffUMA Armadura^000000 que você deseja encantar...";
		next;
		mes "[Artesão Aprendiz]";
		mes "Não me responsabilizo pelo que pode acontecer se você tiver mais de uma do mesmo tipo no seu inventário.";
		next;
		switch(select("Armadura sem slot.:Armadura com slot.:Armadura de alto nível.:Talvez na próxima.")) {
		case 1:
			setarray .@Items[0],2307,2309,2314,2316,2321,2325,2327,2328,2330,2332,2334,2335,2341,2344,2346,2348,2350,2337,2386,2394,2395,2396;
			set .@j,50;
			break;
		case 2:
			setarray .@Items[0],2311,2318,2319,2320,2308,2310,2315,2317,2322,2324,2326,2331,2333,2336,2342,2345,2347,2349,2351;
			set .@j,55;
			set .@k,1;
			break;
		case 3:
			setarray .@Items[0],2364,2365,2391,2374,2375,2376,2377,2378,2379,2380,2381,2382,2387,2388,2389,2390;
			set .@j,60;
			break;
		case 4:
			mes "[Artesão Aprendiz]";
			mes "Por favor, volte quando tiver interesse em encantar sua armadura.";
			close;
		}
		set .@menu$,"";
		for(set .@i,0; .@i<getarraysize(.@Items); set .@i,.@i+1)
			set .@menu$, .@menu$+getitemname(.@Items[.@i])+((.@k)?" [1]":"")+":";
		callsub S_EnchantArmor, .@Items[select(.@menu$)-1], .@j;
		end;
	}
	mes "Eu sou responsável por Encantar Armaduras. Simplificando, tenho estudado formas de fortalecer armaduras.";
	next;
	mes "[Artesão Aprendiz]";
	mes "Se por acaso você quiser encantar sua armadura, traga-me 400.000 zenys e a armadura que deseja encantar e está tudo pronto.";
	close;

S_EnchantArmor:
	set .@itemid,getarg(0);
	set .@failrate,getarg(1);
	mes "[Artesão Aprendiz]";
	if (countitem(.@itemid) == 1) {
		mes "O encantamento de slot custará 400.000 zenys. E será encantada uma opção aleatória. É claro que há uma chance de sua armadura quebrar.";
		next;
		mes "[Artesão Aprendiz]";
		mes "Primeiro e mais importante.";
		mes "^ff5555O Nível de Refino atual da Armadura";
		mes "e as Cartas serão REMOVIDAS.^000000";
		mes "Você ainda quer tentar um Encantamento?";
		next;
		if(select("Hmm... Deixe-me pensar um pouco.:Pode prosseguir.") == 1) {
			mes "[Artesão Aprendiz]";
			mes "Bem, não posso te culpar. Segurança em primeiro lugar, hein?";
			mes "Tenha um bom dia.";
			close;
		}
		mes "[Artesão Aprendiz]";
		mes "Um tanto aventureiro, hein? Bem, vamos começar?";
		close2;
		specialeffect2 EF_MAPPILLAR;
		if (Zeny < 400000) {
			mes "[Artesão Aprendiz]";
			mes "Desculpe, mas você não tem zenys suficientes.";
			close;
		}
		progressbar "ffff00",7;
		set Zeny, Zeny-400000;
		delitem .@itemid,1;
		switch (rand(1,.@failrate)) {
			case 1: set .@addpart,4702;break;
			case 2: set .@addpart,4712;break;
			case 3: set .@addpart,4722;break;
			case 4: set .@addpart,4732;break;
			case 5: set .@addpart,4742;break;
			case 6: set .@addpart,4752;break;
			case 7:
			case 8: set .@addpart,4701;break;
			case 9:
			case 10: set .@addpart,4711;break;
			case 11:
			case 12: set .@addpart,4721;break;
			case 13:
			case 14: set .@addpart,4731;break;
			case 15:
			case 16: set .@addpart,4741;break;
			case 17:
			case 18: set .@addpart,4751;break;
			case 19:
			case 20:
			case 21: set .@addpart,4700;break;
			case 22:
			case 23:
			case 24: set .@addpart,4710;break;
			case 25:
			case 26:
			case 27: set .@addpart,4720;break;
			case 28:
			case 29:
			case 30: set .@addpart,4730;break;
			case 31:
			case 32:
			case 33: set .@addpart,4740;break;
			case 34:
			case 35:
			case 36: set .@addpart,4750;break;
			default:
				specialeffect2 EF_PHARMACY_FAIL;
				mes "[Artesão Aprendiz]";
				mes "Bem, isso é uma pena.";
				mes "O encantamento do equipamento solicitado falhou.";
				close;
		}
		getitem2 .@itemid, 1, 1, 0, 0, 0, 0, 0, .@addpart;
	}
	else {
		mes "Hmm? Não há nada para ser encantado!";
		mes "Por favor, volte com apenas UM equipamento para ser encantado.";
		close;
	}
}
