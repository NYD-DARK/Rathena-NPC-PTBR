alde_alche,24,188,3	script	Guild Dealer	740,{
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais com vocę. -";
		mes "- Por favor, volte novamente -";
		mes "- depois de guardar alguns itens no armazém da Kafra. -";
		close;
	}
	mes "[Gever Al Sharp]";
	mes "Bem-vindo ao";
	mes "Sindicato dos Alquimistas.";
	mes "Como posso ajudá-lo hoje?";
	next;
	switch(select("Comprar materiais.:Comprar um manual de produçăo.:Sair.")) {
	case 1:
		mes "[Gever Al Sharp]";
		mes "O que vocę gostaria?";
		next;
		if (select("Tigela de Mistura - 8 Zeny:Cancelar.") == 1) {
			mes "[Gever Al Sharp]";
			mes "Quantos vocę quer?";
			mes "Digite '0' se quiser sair.";
			next;
			while(1) {
				input .@input,0,2001;
				if (.@input == 0) {
					mes "[Gever Al Sharp]";
					mes "O negócio foi cancelado.";
					mes "Volte na próxima vez.";
					close;
				}
				else if (.@input < 1 || .@input > 2000) {
					mes "[Gever Al Sharp]";
					mes "O número deve";
					mes "ser menor que 2000.";
					next;
				}
				else {
					break;
				}
			}
			set .@sell,.@input * 8;
			set .@item_weight,.@input * 10;
			if (Zeny < .@sell) {
				mes "[Gever Al Sharp]";
				mes "Vocę năo";
				mes "tem zeny suficiente.";
				mes "Verifique quanto zeny";
				mes "vocę tem primeiro.";
				close;
			}
			if ((MaxWeight - Weight) < .@item_weight) {
				mes "[Gever Al Sharp]";
				mes "Năo parece que";
				mes "vocę pode carregar tudo.";
				mes "Por favor, verifique o espaço";
				mes "no seu inventário.";
				close;
			}
			set Zeny, Zeny-.@sell;
			getitem 7134,.@input; //Medicine_Bowl
			mes "[Gever Al Sharp]";
			mes "Obrigado.";
			mes "Volte sempre.";
			close;
		}
		mes "[Gever Al Sharp]";
		mes "Bem entăo,";
		mes "volte quando";
		mes "vocę precisar comprar";
		mes "materiais relacionados ŕ";
		mes "Alquimia, está bem?";
		close;
	case 2:
		mes "[Gever Al Sharp]";
		mes "O que vocę precisa?";
		mes "Manuais geralmente custam 100.000 zenys. Mas há alguns manuais especiais que custam mais.";
		next;
		switch(select("Guia de Criaçăo de Poçőes:Guia de Criaçăo de Álcool:Guia de Criaçăo de Bomba de Fogo:Guia de Criaçăo de Frasco de Ácido:Guia de Criaçăo de Frasco de Planta:Guia de Criaçăo de Frasco de Esfera Marinha:Guia de Criaçăo de Revestimento Total:Condensed Guia de Criaçăo de Poçőes:Cancelar negócio.")) {
		case 1:
			callsub S_SellManual,7144,100000; //Normal_Potion_Book
			break;
		case 2:
			callsub S_SellManual,7127,100000; //Alcol_Create_Book
			break;
		case 3:
			callsub S_SellManual,7128,100000; //FireBottle_Create_Book
			break;
		case 4:
			callsub S_SellManual,7129,100000; //Acid_Create_Book
			break;
		case 5:
			callsub S_SellManual,7130,100000; //Plant_Create_Book
			break;
		case 6:
			callsub S_SellManual,7131,100000; //Mine_Create_Book
			break;
		case 7:
			callsub S_SellManual,7132,100000; //Coating_Create_Book
			break;
		case 8:
			callsub S_SellManual,7133,240000; //Slim_Potion_Create_Book
			break;
		case 9:
			mes "[Gever Al Sharp]";
			mes "Bem entăo...";
			mes "Volte se";
			mes "algum dia precisar comprar";
			mes "um manual de produçăo.";
			close;
		}
	case 3:
		mes "[Gever Al Sharp]";
		mes "Está bem entăo,";
		mes "tenha um bom dia.";
		close;
	}

S_SellManual:
	.@item_id = getarg(0);
	.@zeny_req = getarg(1);

	if (Zeny < .@zeny_req) {
		mes "[Gever Al Sharp]";
		mes "Vocę năo";
		mes "tem zeny suficiente.";
		mes "Verifique quanto zeny";
		mes "vocę tem primeiro.";
		close;
	}
	mes "[Gever Al Sharp]";
	mes getitemname(.@item_id) +"?";
	mes "Isso custará "+ .@zeny_req +" zeny.";
	next;
	if (select("Comprar.:Sair.") == 1) {
		Zeny = Zeny - .@zeny_req;
		getitem .@item_id,1;
		mes "[Gever Al Sharp]";
		mes "Obrigado por";
		mes "sua preferęncia.";
		close;
	}
	mes "[Gever Al Sharp]";
	mes "Volte";
	mes "na próxima vez.";
	close;
}
