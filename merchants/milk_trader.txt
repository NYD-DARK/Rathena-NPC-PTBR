prontera,73,140,0	script	Milk Vendor	86,{
	if (checkweight(1201,1) == 0) {
		mes "^3355FFEspere um minuto!";
		mes "Não posso oferecer nenhum dos meus";
		mes "serviços a você porque";
		mes "você está carregando muita";
		mes "coisa. Coloque seus itens extras no";
		mes "Armazém da Kafra e volte mais tarde~";
		close;
	}
	mes "[Vendedor de Leite]";
	mes "Ei, ei...";
	mes "Se você me trouxer";
	mes "1 Garrafa Vazia e";
	mes "15 Zenys, eu trocarei";
	mes "eles por 1 Leite. O que";
	mes "acha disso?";
	next;
	if (select("Exchange all empty bottles.:Cancel") == 1) {
		if (countitem(713) <= 0) {
			mes "[Vendedor de Leite]";
			mes "Ei...";
			mes "Você não tem";
			mes "nenhuma Garrafa Vazia.";
			mes "Eu não posso te dar";
			mes "esse leite de nenhuma";
			mes "outra forma, sabe...";
			close;
		}
		set .@bottles,countitem(713);
		set .@total_weight,.@bottles * 50;
		set .@total_cost,.@bottles * 15;
		if (Zeny < .@Total_cost) {
			mes "[Vendedor de Leite]";
			mes "Ah, nossa~!";
			mes "Você não tem Zenys";
			mes "suficientes para trocar todas";
			mes "essas Garrafas Vazias por";
			mes "Leite. Você precisa ter";
			mes "at least " + .@total_cost + " zeny.";
			close;
		}
		if (MaxWeight - Weight < .@total_weight) {
			mes "[Vendedor de Leite]";
			mes "Hmm...";
			mes "Você poderia liberar";
			mes "um pouco mais de espaço";
			mes "no seu inventário";
			mes "antes que eu te entregue";
			mes "todo esse leite?";
			close;
		}
		set Zeny, Zeny-.@total_cost;
		delitem 713,.@bottles; //Empty Bottles
		getitem 519,.@bottles; //Milk
		close;
	}
	close;
}
